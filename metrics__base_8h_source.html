99 static void update_min(std::atomic<uint64_t>& min_value, uint64_t new_value,
100 std::memory_order order = std::memory_order_relaxed) noexcept
101 {
102 uint64_t current = min_value.load(std::memory_order_relaxed);
103 while (new_value < current
104 && !min_value.compare_exchange_weak(current, new_value, order,
105 std::memory_order_relaxed))
106 {
107 }
108 }
109
120 static void update_max(std::atomic<uint64_t>& max_value, uint64_t new_value,
121 std::memory_order order = std::memory_order_relaxed) noexcept
122 {
123 uint64_t current = max_value.load(std::memory_order_relaxed);
124 while (new_value > current
125 && !max_value.compare_exchange_weak(current, new_value, order,
126 std::memory_order_relaxed))
127 {
128 }
129 }
130
141 static void update_min_max(std::atomic<uint64_t>& min_value,
142 std::atomic<uint64_t>& max_value,
143 uint64_t new_value) noexcept
144 {
145 update_min(min_value, new_value);
146 update_max(max_value, new_value);

203 [[nodiscard]] static double calculate_rate(uint64_t numerator, uint64_t denominator,
204 double default_value = 100.0) noexcept
