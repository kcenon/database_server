119 static void update_min(std::atomic<uint64_t>& min_value, uint64_t new_value,
120 std::memory_order order = std::memory_order_relaxed) noexcept
121 {
122 uint64_t current = min_value.load(std::memory_order_relaxed);
123 while (new_value < current
124 && !min_value.compare_exchange_weak(current, new_value, order,
125 std::memory_order_relaxed))

127 }
128 }

140 static void update_max(std::atomic<uint64_t>& max_value, uint64_t new_value,

142 {
143 uint64_t current = max_value.load(std::memory_order_relaxed);
144 while (new_value > current
145 && !max_value.compare_exchange_weak(current, new_value, order,

148 }
