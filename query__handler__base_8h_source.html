270template <typename T>
271concept QueryHandler = requires(T handler, const query_request& req,
272 const handler_context& ctx, query_type type) {
273 { handler.handle(req, ctx) } -> std::same_as<query_response>;
274 { handler.can_handle(type) } -> std::same_as<bool>;
275};

284template <typename Handler, typename... Args>
285[[nodiscard]] std::unique_ptr<i_query_handler> make_handler(Args&&... args)
286{
287 return std::make_unique<query_handler_wrapper<Handler>>(
