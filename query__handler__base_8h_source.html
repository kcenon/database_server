278template <typename T>
279concept QueryHandler = requires(T handler, const query_request& req,
280 const handler_context& ctx, query_type type) {
281 { handler.handle(req, ctx) } -> std::same_as<query_response>;
282 { handler.can_handle(type) } -> std::same_as<bool>;
