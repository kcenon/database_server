44class health_monitor_job : public kcenon::common::interfaces::IJob
45{
46public:
47 explicit health_monitor_job(connection_health_monitor* monitor)
48 : monitor_(monitor)
49 {
50 }

53 {
54 if (monitor_)

60
61 std::string get_name() const override { return "health_monitor_loop"; }
62 int get_priority() const override { return 0; }
63
64private:
65 connection_health_monitor* monitor_;
66};

85void connection_health_monitor::start_monitoring()
86{
87 if (is_monitoring_.exchange(true))
88 {
89 return; // Already monitoring
90 }
91
92 stop_requested_ = false;
93
94 if (executor_)
95 {
96 // Use IExecutor for background task
97 auto job = std::make_unique<health_monitor_job>(this);
98 auto result = executor_->execute(std::move(job));
99 if (result.is_ok())
100 {
101 monitoring_future_ = std::move(result.unwrap());
102 }
103 }
104 else
105 {
106 // Fallback to std::async if no executor provided
107 monitoring_future_ = std::async(std::launch::async, [this] { monitoring_loop(); });
108 }
109}

111void connection_health_monitor::stop_monitoring()
112{
113 if (!is_monitoring_.exchange(false))
114 {
115 return; // Not monitoring
116 }
117
118 stop_requested_ = true;
119
120 // Wait for monitoring task to complete
121 if (monitoring_future_.valid())
122 {
123 // Use wait_for with timeout to avoid indefinite blocking
124 auto status = monitoring_future_.wait_for(std::chrono::seconds(5));
125 if (status == std::future_status::timeout)
126 {
127 // Monitoring loop should have exited, but continue anyway
128 }
129 }
130}

132kcenon::common::Result<health_status> connection_health_monitor::check_now()
133{
134 if (!backend_ || !backend_->is_initialized())
135 {
136 std::lock_guard<std::mutex> lock(mutex_);
137 current_status_.is_healthy = false;
138 current_status_.health_score = 0;
139 current_status_.status_message = "Backend not initialized";
140 current_status_.last_check_time = std::chrono::system_clock::now();
141
142 return kcenon::common::error_info{
143 -1, "Backend not initialized", "connection_health_monitor"
144 };
145 }
146
147 auto start = std::chrono::high_resolution_clock::now();
148 auto result = execute_heartbeat();
149 auto end = std::chrono::high_resolution_clock::now();
150
151 auto latency = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
152
153 std::lock_guard<std::mutex> lock(mutex_);
154
155 if (result.is_err())
156 {
157 consecutive_failures_++;
158 consecutive_successes_ = 0;
159 failed_queries_++;
160 total_queries_++;
161
162 current_status_.is_healthy = false;
163 current_status_.status_message = result.error().message;
164 }
165 else
166 {
167 consecutive_successes_++;
168 consecutive_failures_ = 0;
169 successful_queries_++;
170 total_queries_++;
171
172 update_latency_average(latency);
173
174 current_status_.is_healthy = true;
175 current_status_.latency = latency;
176 current_status_.status_message = "Connection healthy";
177 }
178
179 current_status_.successful_queries = successful_queries_;
180 current_status_.failed_queries = failed_queries_;
181 current_status_.health_score = calculate_health_score();
182 current_status_.last_check_time = std::chrono::system_clock::now();
183
184 return current_status_;
185}

230bool connection_health_monitor::predict_failure() const
231{
232 std::lock_guard<std::mutex> lock(mutex_);
233
234 // Predict failure if:
235 // 1. Consecutive failures approaching threshold
236 if (consecutive_failures_ >= config_.failure_threshold - 1)
237 {
238 return true;
239 }
240
241 // 2. Health score declining trend (below 60)
242 if (current_status_.health_score < 60)
243 {
244 return true;
245 }
246
247 // 3. Recent latency spike (> 2x average)
248 if (!latency_history_.empty())
249 {
250 auto avg_latency = std::accumulate(
251 latency_history_.begin(),
252 latency_history_.end(),
253 std::chrono::milliseconds(0))
254 / latency_history_.size();
255
256 if (current_status_.latency > avg_latency * 2)
257 {
258 return true;
259 }
260 }
261
262 return false;
263}

280kcenon::common::VoidResult connection_health_monitor::execute_heartbeat()
281{
282 if (!backend_)
283 {
284 return kcenon::common::error_info{
285 -1, "Backend is null", "connection_health_monitor"
286 };
287 }
288
289 // Execute simple SELECT query as heartbeat
290 // Most databases support "SELECT 1" as a connectivity check
291 auto result = backend_->select_query("SELECT 1");
292
293 if (result.is_err())
294 {
295 return result.error();
296 }
297
298 return kcenon::common::ok();
299}

301uint32_t connection_health_monitor::calculate_health_score() const
302{
303 // Calculate health score (0-100) based on multiple factors
304
305 // Factor 1: Success rate (40% weight)
306 uint64_t total = total_queries_.load();
307 double success_rate
308 = total > 0 ? (static_cast<double>(successful_queries_) / total) : 1.0;
309 uint32_t success_score = static_cast<uint32_t>(success_rate * 40);
310
311 // Factor 2: Latency performance (30% weight)
312 // Assume < 10ms = excellent, 10-50ms = good, 50-100ms = fair, > 100ms = poor
313 uint32_t latency_score = 30;
314 if (!latency_history_.empty())
315 {
316 auto avg_latency = std::accumulate(
317 latency_history_.begin(),
318 latency_history_.end(),
319 std::chrono::milliseconds(0))
320 / latency_history_.size();
321
322 if (avg_latency.count() < 10)
323 {
324 latency_score = 30;
325 }
326 else if (avg_latency.count() < 50)
327 {
328 latency_score = 25;
329 }
330 else if (avg_latency.count() < 100)
331 {
332 latency_score = 15;
333 }
334 else
335 {
336 latency_score = 5;
337 }
338 }
339
340 // Factor 3: Consecutive success streak (20% weight)
341 uint32_t streak_score = std::min(consecutive_successes_.load(), 10u) * 2;
342
343 // Factor 4: Connection uptime (10% weight)
344 auto uptime = std::chrono::system_clock::now() - connection_start_time_;
345 auto uptime_minutes
346 = std::chrono::duration_cast<std::chrono::minutes>(uptime).count();
347 uint32_t uptime_score
348 = std::min(static_cast<uint32_t>(uptime_minutes / 6), 10u); // Max at 1 hour
349
350 // Penalty for consecutive failures
351 uint32_t failure_penalty = consecutive_failures_.load() * 10;
352
353 uint32_t total_score = success_score + latency_score + streak_score + uptime_score;
354 total_score = (total_score > failure_penalty) ? (total_score - failure_penalty) : 0;
355
356 return std::min(total_score, 100u);
357}

359void connection_health_monitor::update_latency_average(

364 {

369void connection_health_monitor::monitoring_loop()
370{
371 while (!stop_requested_)
372 {
373 if (config_.enable_heartbeat)
374 {
375 check_now();
376 }
377
378 // Sleep for heartbeat interval, checking stop_requested frequently
379 auto sleep_until
380 = std::chrono::steady_clock::now() + config_.heartbeat_interval;
381 while (std::chrono::steady_clock::now() < sleep_until && !stop_requested_)
382 {
383 std::this_thread::sleep_for(std::chrono::milliseconds(100));
384 }
385 }
386}
