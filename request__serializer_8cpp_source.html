63std::shared_ptr<container_module::value_container> query_request::serialize() const
64{
65#if KCENON_WITH_CONTAINER_SYSTEM
66 auto container = std::make_shared<container_module::value_container>();
67 container->set_message_type("query_request");
68
69 // Header
70 container->set("version", static_cast<int>(header.version));
71 container->set("message_id", static_cast<long long>(header.message_id));
72 container->set("timestamp", static_cast<long long>(header.timestamp));
73 container->set("correlation_id", header.correlation_id);
74
75 // Auth token
76 container->set("auth_token", token.token);
77 container->set("client_id", token.client_id);
78 container->set("token_expires", static_cast<long long>(token.expires_at));
79
80 // Query
81 container->set("query_type", static_cast<int>(type));
82 container->set("sql", sql);
83
84 // Options
85 container->set("timeout_ms", static_cast<int>(options.timeout_ms));
86 container->set("read_only", options.read_only);
87 container->set("isolation_level", options.isolation_level);
88 container->set("max_rows", static_cast<int>(options.max_rows));
89 container->set("include_metadata", options.include_metadata);
90
91 // Parameters
92 detail::serialize_params(container, params);
93
94 return container;
95#else
96 return nullptr;
97#endif
98}

241bool query_request::is_valid() const noexcept
242{
243 if (type == query_type::unknown)
244 {
245 return false;
246 }
247
248 if (type != query_type::ping && sql.empty())
249 {
250 return false;
251 }
252
253 return true;
254}
