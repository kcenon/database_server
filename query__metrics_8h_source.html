87struct query_execution_metrics
88{
89 std::atomic<uint64_t> total_queries{0};
90 std::atomic<uint64_t> successful_queries{0};
91 std::atomic<uint64_t> failed_queries{0};
92 std::atomic<uint64_t> timeout_queries{0};
93
94 // Latency statistics (nanoseconds)
95 std::atomic<uint64_t> total_latency_ns{0};
96 std::atomic<uint64_t> min_latency_ns{UINT64_MAX};
97 std::atomic<uint64_t> max_latency_ns{0};
98
99 // Query type breakdown
100 std::atomic<uint64_t> select_queries{0};
101 std::atomic<uint64_t> insert_queries{0};
102 std::atomic<uint64_t> update_queries{0};
103 std::atomic<uint64_t> delete_queries{0};
104 std::atomic<uint64_t> other_queries{0};
105
112 void record_query(uint64_t latency_ns, bool success, bool timeout = false)
113 {
114 total_queries.fetch_add(1, std::memory_order_relaxed);
115 total_latency_ns.fetch_add(latency_ns, std::memory_order_relaxed);
116
117 if (timeout)
118 {
119 timeout_queries.fetch_add(1, std::memory_order_relaxed);
120 }
121 else if (success)
122 {
123 successful_queries.fetch_add(1, std::memory_order_relaxed);
124 }
125 else
126 {
127 failed_queries.fetch_add(1, std::memory_order_relaxed);
128 }
129
130 metrics_utils::update_min_max(min_latency_ns, max_latency_ns, latency_ns);
131 }
132
137 [[nodiscard]] double avg_query_latency_ms() const noexcept
138 {
139 return metrics_utils::average_ns_to_ms(
140 total_latency_ns.load(std::memory_order_relaxed),
141 total_queries.load(std::memory_order_relaxed));
142 }
143
148 [[nodiscard]] double success_rate() const noexcept
149 {
150 return metrics_utils::calculate_rate(
151 successful_queries.load(std::memory_order_relaxed),
152 total_queries.load(std::memory_order_relaxed), 100.0);
153 }
154
158 void reset() noexcept
159 {
160 metrics_utils::reset_counter(total_queries);
161 metrics_utils::reset_counter(successful_queries);
162 metrics_utils::reset_counter(failed_queries);
163 metrics_utils::reset_counter(timeout_queries);
164 metrics_utils::reset_counter(total_latency_ns);
165 metrics_utils::reset_min(min_latency_ns);
166 metrics_utils::reset_counter(max_latency_ns);
167 metrics_utils::reset_counter(select_queries);
168 metrics_utils::reset_counter(insert_queries);
169 metrics_utils::reset_counter(update_queries);
170 metrics_utils::reset_counter(delete_queries);
171 metrics_utils::reset_counter(other_queries);
172 }
173};

182struct cache_performance_metrics
183{
184 std::atomic<uint64_t> cache_hits{0};
185 std::atomic<uint64_t> cache_misses{0};
186 std::atomic<uint64_t> cache_evictions{0};
187 std::atomic<uint64_t> cache_expirations{0};
188 std::atomic<uint64_t> cache_size_bytes{0};
189 std::atomic<uint64_t> cache_entries{0};
190
194 void record_hit()
195 {
196 cache_hits.fetch_add(1, std::memory_order_relaxed);
197 }
198
202 void record_miss()
203 {
204 cache_misses.fetch_add(1, std::memory_order_relaxed);
205 }
206
210 void record_eviction()
211 {
212 cache_evictions.fetch_add(1, std::memory_order_relaxed);
213 }
214
218 void record_expiration()
219 {
220 cache_expirations.fetch_add(1, std::memory_order_relaxed);
221 }
222
227 [[nodiscard]] double cache_hit_ratio() const noexcept
228 {
229 uint64_t hits = cache_hits.load(std::memory_order_relaxed);
230 uint64_t misses = cache_misses.load(std::memory_order_relaxed);
231 return metrics_utils::calculate_rate(hits, hits + misses, 0.0);
232 }
233
237 void reset() noexcept
238 {
239 metrics_utils::reset_counter(cache_hits);
240 metrics_utils::reset_counter(cache_misses);
241 metrics_utils::reset_counter(cache_evictions);
242 metrics_utils::reset_counter(cache_expirations);
243 // Don't reset cache_size_bytes and cache_entries (they reflect current state)
244 }

356 {

376 }

386 terminated_sessions.load(std::memory_order_relaxed));
387 }
388
392 void reset() noexcept
393 {
394 metrics_utils::reset_counter(total_sessions);
395 metrics_utils::reset_counter(terminated_sessions);
396 metrics_utils::reset_counter(auth_successes);
397 metrics_utils::reset_counter(auth_failures);
398 metrics_utils::reset_counter(total_session_duration_ns);
