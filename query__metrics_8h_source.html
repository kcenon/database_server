65struct query_execution_metrics
66{
67 std::atomic<uint64_t> total_queries{0};
68 std::atomic<uint64_t> successful_queries{0};
69 std::atomic<uint64_t> failed_queries{0};
70 std::atomic<uint64_t> timeout_queries{0};
71
72 // Latency statistics (nanoseconds)
73 std::atomic<uint64_t> total_latency_ns{0};
74 std::atomic<uint64_t> min_latency_ns{UINT64_MAX};
75 std::atomic<uint64_t> max_latency_ns{0};
76
77 // Query type breakdown
78 std::atomic<uint64_t> select_queries{0};
79 std::atomic<uint64_t> insert_queries{0};
80 std::atomic<uint64_t> update_queries{0};
81 std::atomic<uint64_t> delete_queries{0};
82 std::atomic<uint64_t> other_queries{0};
83
90 void record_query(uint64_t latency_ns, bool success, bool timeout = false)
91 {
92 total_queries.fetch_add(1, std::memory_order_relaxed);
93 total_latency_ns.fetch_add(latency_ns, std::memory_order_relaxed);
94
95 if (timeout)
96 {
97 timeout_queries.fetch_add(1, std::memory_order_relaxed);
98 }
99 else if (success)
100 {
101 successful_queries.fetch_add(1, std::memory_order_relaxed);
102 }
103 else
104 {
105 failed_queries.fetch_add(1, std::memory_order_relaxed);
106 }
107
108 metrics_utils::update_min_max(min_latency_ns, max_latency_ns, latency_ns);
109 }
110
115 [[nodiscard]] double avg_query_latency_ms() const noexcept
116 {
117 return metrics_utils::average_ns_to_ms(
118 total_latency_ns.load(std::memory_order_relaxed),
119 total_queries.load(std::memory_order_relaxed));
120 }
121
126 [[nodiscard]] double success_rate() const noexcept
127 {
128 return metrics_utils::calculate_rate(
129 successful_queries.load(std::memory_order_relaxed),
130 total_queries.load(std::memory_order_relaxed), 100.0);
131 }
132
136 void reset() noexcept
137 {
138 metrics_utils::reset_counter(total_queries);
139 metrics_utils::reset_counter(successful_queries);
140 metrics_utils::reset_counter(failed_queries);
141 metrics_utils::reset_counter(timeout_queries);
142 metrics_utils::reset_counter(total_latency_ns);
143 metrics_utils::reset_min(min_latency_ns);
144 metrics_utils::reset_counter(max_latency_ns);
145 metrics_utils::reset_counter(select_queries);
146 metrics_utils::reset_counter(insert_queries);
147 metrics_utils::reset_counter(update_queries);
148 metrics_utils::reset_counter(delete_queries);
149 metrics_utils::reset_counter(other_queries);
150 }
151};

160struct cache_performance_metrics
161{
162 std::atomic<uint64_t> cache_hits{0};
163 std::atomic<uint64_t> cache_misses{0};
164 std::atomic<uint64_t> cache_evictions{0};
165 std::atomic<uint64_t> cache_expirations{0};
166 std::atomic<uint64_t> cache_size_bytes{0};
167 std::atomic<uint64_t> cache_entries{0};
168
172 void record_hit()
173 {
174 cache_hits.fetch_add(1, std::memory_order_relaxed);
175 }
176
180 void record_miss()
181 {
182 cache_misses.fetch_add(1, std::memory_order_relaxed);
183 }
184
188 void record_eviction()
189 {
190 cache_evictions.fetch_add(1, std::memory_order_relaxed);
191 }
192
196 void record_expiration()
197 {
198 cache_expirations.fetch_add(1, std::memory_order_relaxed);
199 }
200
205 [[nodiscard]] double cache_hit_ratio() const noexcept
206 {
207 uint64_t hits = cache_hits.load(std::memory_order_relaxed);
208 uint64_t misses = cache_misses.load(std::memory_order_relaxed);
209 return metrics_utils::calculate_rate(hits, hits + misses, 0.0);
210 }
211
215 void reset() noexcept
216 {
217 metrics_utils::reset_counter(cache_hits);
218 metrics_utils::reset_counter(cache_misses);
219 metrics_utils::reset_counter(cache_evictions);
220 metrics_utils::reset_counter(cache_expirations);
221 // Don't reset cache_size_bytes and cache_entries (they reflect current state)
222 }
223};

307struct session_performance_metrics

315 std::atomic<uint64_t> auth_failures{0};
316
317 // Duration statistics (nanoseconds)
318 std::atomic<uint64_t> total_session_duration_ns{0};
319
323 void record_session_start()
324 {
325 active_sessions.fetch_add(1, std::memory_order_relaxed);
326 total_sessions.fetch_add(1, std::memory_order_relaxed);
327 }
328
333 void record_session_end(uint64_t duration_ns)
334 {
335 active_sessions.fetch_sub(1, std::memory_order_relaxed);
336 terminated_sessions.fetch_add(1, std::memory_order_relaxed);
337 total_session_duration_ns.fetch_add(duration_ns, std::memory_order_relaxed);
338 }
339
344 void record_auth(bool success)
345 {
346 if (success)
347 {
348 auth_successes.fetch_add(1, std::memory_order_relaxed);
349 }
350 else
351 {
352 auth_failures.fetch_add(1, std::memory_order_relaxed);
353 }
354 }
355
360 [[nodiscard]] double avg_session_duration_sec() const noexcept
361 {
362 return metrics_utils::average_ns_to_sec(
363 total_session_duration_ns.load(std::memory_order_relaxed),
364 terminated_sessions.load(std::memory_order_relaxed));
365 }
366
370 void reset() noexcept
371 {
372 metrics_utils::reset_counter(total_sessions);
373 metrics_utils::reset_counter(terminated_sessions);
374 metrics_utils::reset_counter(auth_successes);
375 metrics_utils::reset_counter(auth_failures);
376 metrics_utils::reset_counter(total_session_duration_ns);
377 // Don't reset active_sessions (it reflects current state)
378 }
