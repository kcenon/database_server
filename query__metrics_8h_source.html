63struct query_execution_metrics
64{
65 std::atomic<uint64_t> total_queries{0};
66 std::atomic<uint64_t> successful_queries{0};
67 std::atomic<uint64_t> failed_queries{0};
68 std::atomic<uint64_t> timeout_queries{0};
69
70 // Latency statistics (nanoseconds)
71 std::atomic<uint64_t> total_latency_ns{0};
72 std::atomic<uint64_t> min_latency_ns{UINT64_MAX};
73 std::atomic<uint64_t> max_latency_ns{0};
74
75 // Query type breakdown
76 std::atomic<uint64_t> select_queries{0};
77 std::atomic<uint64_t> insert_queries{0};
78 std::atomic<uint64_t> update_queries{0};
79 std::atomic<uint64_t> delete_queries{0};
80 std::atomic<uint64_t> other_queries{0};
81
88 void record_query(uint64_t latency_ns, bool success, bool timeout = false)
89 {
90 total_queries.fetch_add(1, std::memory_order_relaxed);
91 total_latency_ns.fetch_add(latency_ns, std::memory_order_relaxed);
92
93 if (timeout)
94 {
95 timeout_queries.fetch_add(1, std::memory_order_relaxed);
96 }
97 else if (success)
98 {
99 successful_queries.fetch_add(1, std::memory_order_relaxed);
100 }
101 else
102 {
103 failed_queries.fetch_add(1, std::memory_order_relaxed);
104 }
105
106 // Update min latency
107 uint64_t current_min = min_latency_ns.load(std::memory_order_relaxed);
108 while (latency_ns < current_min
109 && !min_latency_ns.compare_exchange_weak(
110 current_min, latency_ns, std::memory_order_relaxed))
111 {
112 }
113
114 // Update max latency
115 uint64_t current_max = max_latency_ns.load(std::memory_order_relaxed);
116 while (latency_ns > current_max
117 && !max_latency_ns.compare_exchange_weak(
118 current_max, latency_ns, std::memory_order_relaxed))
119 {
120 }
121 }
122
127 [[nodiscard]] double avg_query_latency_ms() const noexcept
128 {
129 uint64_t total = total_queries.load(std::memory_order_relaxed);
130 if (total == 0)
131 return 0.0;
132 return static_cast<double>(total_latency_ns.load(std::memory_order_relaxed))
133 / static_cast<double>(total) / 1000000.0;
134 }
135
140 [[nodiscard]] double success_rate() const noexcept
141 {
142 uint64_t total = total_queries.load(std::memory_order_relaxed);
143 if (total == 0)
144 return 100.0;
145 return static_cast<double>(successful_queries.load(std::memory_order_relaxed))
146 / static_cast<double>(total) * 100.0;
147 }
148
152 void reset() noexcept
153 {
154 total_queries.store(0, std::memory_order_relaxed);
155 successful_queries.store(0, std::memory_order_relaxed);
156 failed_queries.store(0, std::memory_order_relaxed);
157 timeout_queries.store(0, std::memory_order_relaxed);
158 total_latency_ns.store(0, std::memory_order_relaxed);
159 min_latency_ns.store(UINT64_MAX, std::memory_order_relaxed);
160 max_latency_ns.store(0, std::memory_order_relaxed);
161 select_queries.store(0, std::memory_order_relaxed);
162 insert_queries.store(0, std::memory_order_relaxed);
163 update_queries.store(0, std::memory_order_relaxed);
164 delete_queries.store(0, std::memory_order_relaxed);
165 other_queries.store(0, std::memory_order_relaxed);
166 }
167};

176struct cache_performance_metrics
177{
178 std::atomic<uint64_t> cache_hits{0};
179 std::atomic<uint64_t> cache_misses{0};
180 std::atomic<uint64_t> cache_evictions{0};
181 std::atomic<uint64_t> cache_expirations{0};
182 std::atomic<uint64_t> cache_size_bytes{0};
183 std::atomic<uint64_t> cache_entries{0};
184
188 void record_hit()
189 {
190 cache_hits.fetch_add(1, std::memory_order_relaxed);
191 }
192
196 void record_miss()
197 {
198 cache_misses.fetch_add(1, std::memory_order_relaxed);
199 }
200
204 void record_eviction()
205 {
206 cache_evictions.fetch_add(1, std::memory_order_relaxed);
207 }
208
212 void record_expiration()
213 {
214 cache_expirations.fetch_add(1, std::memory_order_relaxed);
215 }
216
221 [[nodiscard]] double cache_hit_ratio() const noexcept
222 {
223 uint64_t hits = cache_hits.load(std::memory_order_relaxed);
224 uint64_t misses = cache_misses.load(std::memory_order_relaxed);
225 uint64_t total = hits + misses;
226 if (total == 0)
227 return 0.0;
228 return static_cast<double>(hits) / static_cast<double>(total) * 100.0;
229 }
230
234 void reset() noexcept
235 {
236 cache_hits.store(0, std::memory_order_relaxed);
237 cache_misses.store(0, std::memory_order_relaxed);
238 cache_evictions.store(0, std::memory_order_relaxed);
239 cache_expirations.store(0, std::memory_order_relaxed);
240 // Don't reset cache_size_bytes and cache_entries (they reflect current state)
241 }
242};

329struct session_performance_metrics
330{
331 std::atomic<uint64_t> active_sessions{0};

333 std::atomic<uint64_t> terminated_sessions{0};
334
335 // Authentication statistics
336 std::atomic<uint64_t> auth_successes{0};
337 std::atomic<uint64_t> auth_failures{0};
338
339 // Duration statistics (nanoseconds)
340 std::atomic<uint64_t> total_session_duration_ns{0};
341
345 void record_session_start()
346 {
347 active_sessions.fetch_add(1, std::memory_order_relaxed);
348 total_sessions.fetch_add(1, std::memory_order_relaxed);
349 }
350
355 void record_session_end(uint64_t duration_ns)
356 {
357 active_sessions.fetch_sub(1, std::memory_order_relaxed);
358 terminated_sessions.fetch_add(1, std::memory_order_relaxed);
359 total_session_duration_ns.fetch_add(duration_ns, std::memory_order_relaxed);
360 }
361
366 void record_auth(bool success)
367 {
368 if (success)
369 {
370 auth_successes.fetch_add(1, std::memory_order_relaxed);
371 }
372 else
373 {
374 auth_failures.fetch_add(1, std::memory_order_relaxed);
375 }
376 }
377
382 [[nodiscard]] double avg_session_duration_sec() const noexcept
383 {
384 uint64_t terminated = terminated_sessions.load(std::memory_order_relaxed);
385 if (terminated == 0)
386 return 0.0;
387 return static_cast<double>(
388 total_session_duration_ns.load(std::memory_order_relaxed))
389 / static_cast<double>(terminated) / 1000000000.0;
390 }
391
395 void reset() noexcept
396 {
397 total_sessions.store(0, std::memory_order_relaxed);
398 terminated_sessions.store(0, std::memory_order_relaxed);
399 auth_successes.store(0, std::memory_order_relaxed);
400 auth_failures.store(0, std::memory_order_relaxed);
401 total_session_duration_ns.store(0, std::memory_order_relaxed);
402 // Don't reset active_sessions (it reflects current state)
403 }
404};
