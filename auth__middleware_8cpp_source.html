45 std::chrono::duration_cast<std::chrono::milliseconds>(

56bool auth_validator::needs_refresh(const auth_token& token,
57 uint32_t refresh_window_ms) const
58{
59 if (token.expires_at == 0)
60 {
61 return false;
62 }
63
64 auto now = current_timestamp_ms();
65 auto refresh_threshold = token.expires_at - refresh_window_ms;
66 return now >= refresh_threshold;
67}

73auth_result simple_token_validator::validate(const auth_token& token)
74{
75 auth_result result;
76
77 if (token.token.empty())
78 {
79 result.success = false;
80 result.code = status_code::authentication_failed;
81 result.message = "Token is empty";
82 return result;
83 }
84
85 if (token.is_expired())
86 {
87 result.success = false;
88 result.code = status_code::authentication_failed;
89 result.message = "Token has expired";
90 return result;
91 }
92
93 result.success = true;
94 result.code = status_code::ok;
95 result.client_id = token.client_id;
96 return result;
97}

108bool rate_limiter::allow_request(const std::string& client_id)
109{
110 if (!config_.enabled)
111 {
112 return true;
113 }
114
115 std::lock_guard<std::mutex> lock(entries_mutex_);
116
117 auto now = current_timestamp_ms();
118 auto& entry = entries_[client_id];
119
120 entry.total_requests.fetch_add(1, std::memory_order_relaxed);
121
122 // Check if blocked
123 if (entry.blocked_until > now)
124 {
125 entry.rejected_requests.fetch_add(1, std::memory_order_relaxed);
126 return false;
127 }
128
129 // Clean old timestamps outside the window
130 auto window_start = now - config_.window_size_ms;
131 entry.request_timestamps.erase(
132 std::remove_if(entry.request_timestamps.begin(),
133 entry.request_timestamps.end(),
134 [window_start](uint64_t ts) { return ts < window_start; }),
135 entry.request_timestamps.end());
136
137 // Check if within limits
138 auto requests_in_window = static_cast<uint32_t>(entry.request_timestamps.size());
139 auto max_requests = config_.requests_per_second * config_.window_size_ms / 1000;
140
141 // Allow burst
142 if (requests_in_window >= std::max(max_requests, config_.burst_size))
143 {
144 entry.blocked_until = now + config_.block_duration_ms;
145 entry.rejected_requests.fetch_add(1, std::memory_order_relaxed);
146 return false;
147 }
148
149 entry.request_timestamps.push_back(now);
150 return true;
151}

153uint32_t rate_limiter::remaining_requests(const std::string& client_id) const
154{
155 if (!config_.enabled)
156 {
157 return UINT32_MAX;
158 }
159
160 std::lock_guard<std::mutex> lock(entries_mutex_);
161
162 auto it = entries_.find(client_id);
163 if (it == entries_.end())
164 {
165 return config_.requests_per_second * config_.window_size_ms / 1000;
166 }
167
168 auto now = current_timestamp_ms();
169 auto window_start = now - config_.window_size_ms;
170
171 auto requests_in_window = std::count_if(
172 it->second.request_timestamps.begin(),
173 it->second.request_timestamps.end(),
174 [window_start](uint64_t ts) { return ts >= window_start; });
175
176 auto max_requests = config_.requests_per_second * config_.window_size_ms / 1000;
177 if (static_cast<uint32_t>(requests_in_window) >= max_requests)
178 {
179 return 0;
180 }
181
182 return max_requests - static_cast<uint32_t>(requests_in_window);
183}

185bool rate_limiter::is_blocked(const std::string& client_id) const
186{
187 if (!config_.enabled)
188 {
189 return false;
190 }
191
192 std::lock_guard<std::mutex> lock(entries_mutex_);
193
194 auto it = entries_.find(client_id);
195 if (it == entries_.end())
196 {
197 return false;
198 }
199

203uint64_t rate_limiter::block_expires_at(const std::string& client_id) const
204{
205 std::lock_guard<std::mutex> lock(entries_mutex_);
206
207 auto it = entries_.find(client_id);
208 if (it == entries_.end())
209 {
210 return 0;
211 }
212
213 auto now = current_timestamp_ms();
214 if (it->second.blocked_until <= now)
215 {
216 return 0;
217 }
218
219 return it->second.blocked_until;
220}

228void rate_limiter::cleanup()
229{
230 std::lock_guard<std::mutex> lock(entries_mutex_);
231
232 auto now = current_timestamp_ms();
233 auto window_start = now - config_.window_size_ms;
234
235 for (auto it = entries_.begin(); it != entries_.end();)
236 {
237 // Remove entry if not blocked and no recent requests
238 bool has_recent_requests = std::any_of(
239 it->second.request_timestamps.begin(),
240 it->second.request_timestamps.end(),
241 [window_start](uint64_t ts) { return ts >= window_start; });
242
243 bool is_blocked = it->second.blocked_until > now;
244
245 if (!has_recent_requests && !is_blocked)
246 {
247 it = entries_.erase(it);
248 }
249 else
250 {
251 ++it;
252 }
253 }

265double auth_metrics::success_rate() const noexcept
266{
267 auto total = total_auth_attempts.load(std::memory_order_relaxed);
268 if (total == 0)
269 {
270 return 100.0;
271 }
272
273 auto successful = successful_auths.load(std::memory_order_relaxed);
274 return (static_cast<double>(successful) / static_cast<double>(total)) * 100.0;
275}
276
277double auth_metrics::rate_limit_rejection_rate() const noexcept
278{
279 auto total = total_auth_attempts.load(std::memory_order_relaxed);
280 if (total == 0)
281 {
282 return 0.0;
283 }
284
285 auto rejected = rate_limited_requests.load(std::memory_order_relaxed);
286 return (static_cast<double>(rejected) / static_cast<double>(total)) * 100.0;
287}

321auth_result auth_middleware::authenticate(const std::string& session_id,
322 const auth_token& token)
323{
324 metrics_.total_auth_attempts.fetch_add(1, std::memory_order_relaxed);
325
326 if (!auth_config_.enabled)
327 {
328 auth_result result;
329 result.success = true;
330 result.code = status_code::ok;
331 result.client_id = token.client_id;
332 metrics_.successful_auths.fetch_add(1, std::memory_order_relaxed);
333 return result;
334 }
335
336 auto result = validator_->validate(token);
337
338 if (result.success)
339 {
340 metrics_.successful_auths.fetch_add(1, std::memory_order_relaxed);
341 emit_event(auth_event_type::auth_success, token.client_id, session_id);
342
343 // Update session-client mapping
344 {
345 std::lock_guard<std::mutex> lock(sessions_mutex_);
346 session_client_map_[session_id] = token.client_id;
347 }
348 }
349 else
350 {
351 metrics_.failed_auths.fetch_add(1, std::memory_order_relaxed);
352
353 if (token.is_expired())
354 {
355 metrics_.expired_tokens.fetch_add(1, std::memory_order_relaxed);
356 emit_event(auth_event_type::token_expired, token.client_id, session_id,
357 result.message);
358 }
359 else
360 {
361 metrics_.invalid_tokens.fetch_add(1, std::memory_order_relaxed);
362 emit_event(auth_event_type::token_invalid, token.client_id, session_id,
363 result.message);
364 }
365 }
366
367 return result;
368}

370bool auth_middleware::check_rate_limit(const std::string& client_id)
371{
372 bool allowed = rate_limiter_.allow_request(client_id);
373
374 if (!allowed)
375 {
376 metrics_.rate_limited_requests.fetch_add(1, std::memory_order_relaxed);
377 emit_event(auth_event_type::rate_limited, client_id, "",
378 "Rate limit exceeded");
379 }
380
381 return allowed;
382}

384auth_result auth_middleware::check(const std::string& session_id,
385 const auth_token& token)
386{
387 // First check rate limit
388 if (!check_rate_limit(token.client_id.empty() ? session_id : token.client_id))
389 {
390 auth_result result;
391 result.success = false;
392 result.code = status_code::rate_limited;
393 result.message = "Rate limit exceeded";
394 return result;
395 }
396
397 // Then authenticate
398 return authenticate(session_id, token);
399}

412void auth_middleware::on_session_destroyed(const std::string& session_id)
413{
414 std::string client_id;
415 {
416 std::lock_guard<std::mutex> lock(sessions_mutex_);
417 if (auto it = session_client_map_.find(session_id);
418 it != session_client_map_.end())
419 {
420 client_id = it->second;
421 session_client_map_.erase(it);
422 }
423 }
424
425 emit_event(auth_event_type::session_destroyed, client_id, session_id);
426}

454void auth_middleware::emit_event(auth_event_type type,
455 const std::string& client_id,
456 const std::string& session_id,
457 const std::string& details)
458{
459 audit_callback_t callback;

464
