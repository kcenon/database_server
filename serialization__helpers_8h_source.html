87template<typename VariantT>
88void serialize_variant_value(std::shared_ptr<container_module::value_container>& container,
89 const std::string& prefix,
90 const VariantT& value)
91{
92 container->set(prefix + "type", static_cast<int>(value.index()));
93
94 std::visit(
95 [&container, &prefix](auto&& arg)
96 {
97 using T = std::decay_t<decltype(arg)>;
98 if constexpr (std::is_same_v<T, std::monostate>)
99 {
100 // NULL value - nothing to store
101 }
102 else if constexpr (std::is_same_v<T, bool>)
103 {
104 container->set(prefix + "value_bool", arg);
105 }
106 else if constexpr (std::is_same_v<T, int64_t>)
107 {
108 container->set(prefix + "value_int", static_cast<long long>(arg));
109 }
110 else if constexpr (std::is_same_v<T, double>)
111 {
112 container->set(prefix + "value_double", arg);
113 }
114 else if constexpr (std::is_same_v<T, std::string>)
115 {
116 container->set(prefix + "value_string", arg);
117 }
118 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
119 {
120 container->set(prefix + "value_bytes", arg);
121 }
122 },
123 value);
124}

129template<typename VariantT>
130VariantT deserialize_variant_value(
131 const std::shared_ptr<container_module::value_container>& container,
132 const std::string& prefix)
133{
134 int type_tag = 0;
135 if (auto val = container->get(prefix + "type"))
136 {
137 if (std::holds_alternative<int>(val->data))
138 {
139 type_tag = std::get<int>(val->data);
140 }
141 }
142
143 VariantT result;
144 switch (type_tag)
145 {
146 case static_cast<int>(value_type_tag::null_value):
147 result = std::monostate{};
148 break;
149 case static_cast<int>(value_type_tag::bool_value):
150 if (auto val = container->get(prefix + "value_bool"))
151 {
152 if (std::holds_alternative<bool>(val->data))
153 {
154 result = std::get<bool>(val->data);
155 }
156 }
157 break;
158 case static_cast<int>(value_type_tag::int64_value):
159 if (auto val = container->get(prefix + "value_int"))
160 {
161 if (std::holds_alternative<long long>(val->data))
162 {
163 result = static_cast<int64_t>(std::get<long long>(val->data));
164 }
165 }
166 break;
167 case static_cast<int>(value_type_tag::double_value):
168 if (auto val = container->get(prefix + "value_double"))
169 {
170 if (std::holds_alternative<double>(val->data))
171 {
172 result = std::get<double>(val->data);
173 }
174 }
175 break;
176 case static_cast<int>(value_type_tag::string_value):
177 if (auto val = container->get(prefix + "value_string"))
178 {
179 if (std::holds_alternative<std::string>(val->data))
180 {
181 result = std::get<std::string>(val->data);
182 }
183 }
184 break;
185 case static_cast<int>(value_type_tag::bytes_value):
186 if (auto val = container->get(prefix + "value_bytes"))
187 {
188 if (std::holds_alternative<std::vector<uint8_t>>(val->data))
189 {
190 result = std::get<std::vector<uint8_t>>(val->data);
191 }
192 }
193 break;
194 }
195
196 return result;
197}
