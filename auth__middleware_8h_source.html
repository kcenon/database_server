107constexpr const char* to_string(auth_event_type type) noexcept
108{
109 switch (type)
110 {
111 case auth_event_type::auth_success:
112 return "auth_success";
113 case auth_event_type::auth_failure:
114 return "auth_failure";
115 case auth_event_type::token_expired:
116 return "token_expired";
117 case auth_event_type::token_invalid:
118 return "token_invalid";
119 case auth_event_type::rate_limited:
120 return "rate_limited";
121 case auth_event_type::permission_denied:
122 return "permission_denied";
123 case auth_event_type::session_created:
124 return "session_created";
125 case auth_event_type::session_destroyed:
126 return "session_destroyed";
127 default:
128 return "unknown";
129 }
