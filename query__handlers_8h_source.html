86inline pooling::connection_priority get_priority_for_query_type(query_type type)
87{
88 switch (type)
89 {
90 case query_type::select:
91 return pooling::PRIORITY_NORMAL_QUERY;
92 case query_type::insert:
93 case query_type::update:
94 case query_type::del:
95 return pooling::PRIORITY_TRANSACTION;
96 case query_type::execute:
97 return pooling::PRIORITY_TRANSACTION;
98 case query_type::batch:
99 return pooling::PRIORITY_TRANSACTION;
100 case query_type::ping:
101 return pooling::PRIORITY_HEALTH_CHECK;
102 default:
103 return pooling::PRIORITY_NORMAL_QUERY;
104 }
105}
