97inline uint64_t current_timestamp_us()
98{
99 return static_cast<uint64_t>(
100 std::chrono::duration_cast<std::chrono::microseconds>(
101 std::chrono::steady_clock::now().time_since_epoch())
102 .count());

108inline pooling::connection_priority get_priority_for_query_type(query_type type)
109{
110 switch (type)
111 {
112 case query_type::select:
113 return pooling::PRIORITY_NORMAL_QUERY;
114 case query_type::insert:
115 case query_type::update:
116 case query_type::del:
117 return pooling::PRIORITY_TRANSACTION;
118 case query_type::execute:
119 return pooling::PRIORITY_TRANSACTION;
120 case query_type::batch:
121 return pooling::PRIORITY_TRANSACTION;
122 case query_type::ping:
123 return pooling::PRIORITY_HEALTH_CHECK;
124 default:
125 return pooling::PRIORITY_NORMAL_QUERY;
126 }
127}
