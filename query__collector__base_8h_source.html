179 bool initialize(const collector_config& config)
180 {
181 // Parse common configuration
182 if (auto it = config.find("enabled"); it != config.end())
183 {
184 enabled_ = (it->second == "true" || it->second == "1");
185 }
186
187 // Delegate to derived class for specific initialization
188 return derived().do_initialize(config);
189 }
190
195 void collect_query_metrics(const query_execution& exec)
196 {
197 if (!enabled_)
198 {
199 return;
200 }
201
202 try
203 {
204 derived().do_collect_query(exec);
205 ++collection_count_;
206 }
207 catch (...)
208 {
209 ++collection_errors_;
210 }
211 }

283 [[nodiscard]] std::string get_name() const

292 [[nodiscard]] bool is_healthy() const
293 {
294 if (!enabled_)

305 [[nodiscard]] stats_map get_statistics() const
306 {
307 std::lock_guard<std::mutex> lock(stats_mutex_);
308 stats_map stats;
309
310 // Common statistics
311 stats["enabled"] = enabled_ ? 1.0 : 0.0;
312 stats["collection_count"] = static_cast<double>(collection_count_.load());
313 stats["collection_errors"] = static_cast<double>(collection_errors_.load());
314
315 // Let derived class add specific statistics
316 derived().do_add_statistics(stats);
317
318 return stats;
319 }

325 [[nodiscard]] bool is_enabled() const noexcept
326 {
327 return enabled_;
328 }

334 [[nodiscard]] size_t get_collection_count() const noexcept

343 [[nodiscard]] size_t get_collection_errors() const noexcept

352 [[nodiscard]] const query_server_metrics& get_metrics() const noexcept
353 {
354 return derived().do_get_metrics();
355 }
356
357protected:
358 // Configuration
359 bool enabled_{true};
360
361 // Statistics
362 mutable std::mutex stats_mutex_;
363 std::atomic<size_t> collection_count_{0};
364 std::atomic<size_t> collection_errors_{0};
365
366private:
370 Derived& derived()
371 {
372 return static_cast<Derived&>(*this);
373 }
374
375 const Derived& derived() const
376 {
377 return static_cast<const Derived&>(*this);
378 }

387template <typename T>
388concept QueryCollector = requires(T t, const collector_config& config,
389 const query_execution& exec,
390 const pool_stats& pool,
391 const cache_stats& cache,
392 const session_stats& session,
393 stats_map& stats) {
394 { T::collector_name } -> std::convertible_to<const char*>;
395 { t.do_initialize(config) } -> std::same_as<bool>;
396 { t.do_collect_query(exec) } -> std::same_as<void>;
397 { t.do_collect_pool(pool) } -> std::same_as<void>;
398 { t.do_collect_cache(cache) } -> std::same_as<void>;
399 { t.do_collect_session(session) } -> std::same_as<void>;
400 { t.is_available() } -> std::same_as<bool>;
401 { t.do_add_statistics(stats) } -> std::same_as<void>;
402 { t.do_get_metrics() } -> std::same_as<const query_server_metrics&>;
403};
