152template <typename Derived>
153class query_collector_base
154{
155public:
156 query_collector_base() = default;
157 virtual ~query_collector_base() = default;
158
159 // Non-copyable, non-moveable
160 query_collector_base(const query_collector_base&) = delete;
161 query_collector_base& operator=(const query_collector_base&) = delete;
162 query_collector_base(query_collector_base&&) = delete;
163 query_collector_base& operator=(query_collector_base&&) = delete;

170 bool initialize(const collector_config& config)
171 {
172 // Parse common configuration
173 if (auto it = config.find("enabled"); it != config.end())
174 {
175 enabled_ = (it->second == "true" || it->second == "1");
176 }
177
178 // Delegate to derived class for specific initialization
179 return derived().do_initialize(config);
180 }
181
186 void collect_query_metrics(const query_execution& exec)
187 {
188 if (!enabled_)
189 {
190 return;
191 }
192
193 try
194 {
195 derived().do_collect_query(exec);
196 ++collection_count_;
197 }
198 catch (...)
199 {
200 ++collection_errors_;
201 }
202 }
203
208 void collect_pool_metrics(const pool_stats& stats)
209 {
210 if (!enabled_)
211 {
212 return;
213 }
214
215 try
216 {
217 derived().do_collect_pool(stats);
218 ++collection_count_;
219 }
220 catch (...)
221 {
222 ++collection_errors_;
223 }
224 }
225
230 void collect_cache_metrics(const cache_stats& stats)
231 {
232 if (!enabled_)
233 {
234 return;
235 }
236
237 try
238 {
239 derived().do_collect_cache(stats);
240 ++collection_count_;
241 }
242 catch (...)
243 {
244 ++collection_errors_;
245 }
246 }
247
252 void collect_session_metrics(const session_stats& stats)
253 {
254 if (!enabled_)
255 {
256 return;
257 }
258
259 try
260 {
261 derived().do_collect_session(stats);
262 ++collection_count_;
263 }
264 catch (...)
265 {
266 ++collection_errors_;
267 }
268 }
269
274 [[nodiscard]] std::string get_name() const
275 {
276 return Derived::collector_name;
277 }
278
283 [[nodiscard]] bool is_healthy() const
284 {
285 if (!enabled_)
286 {
287 return true; // Disabled collectors are considered healthy
288 }
289 return derived().is_available();
290 }
291
296 [[nodiscard]] stats_map get_statistics() const
297 {
298 std::lock_guard<std::mutex> lock(stats_mutex_);
299 stats_map stats;
300
301 // Common statistics
302 stats["enabled"] = enabled_ ? 1.0 : 0.0;
303 stats["collection_count"] = static_cast<double>(collection_count_.load());
304 stats["collection_errors"] = static_cast<double>(collection_errors_.load());
305
306 // Let derived class add specific statistics
307 derived().do_add_statistics(stats);
308
309 return stats;
310 }
311
316 [[nodiscard]] bool is_enabled() const noexcept
317 {
318 return enabled_;
319 }
320
325 [[nodiscard]] size_t get_collection_count() const noexcept
326 {
327 return collection_count_.load();
328 }
329
334 [[nodiscard]] size_t get_collection_errors() const noexcept
335 {
336 return collection_errors_.load();
337 }
338
343 [[nodiscard]] const query_server_metrics& get_metrics() const noexcept
344 {
345 return derived().do_get_metrics();
346 }
347
348protected:
349 // Configuration
350 bool enabled_{true};
351
352 // Statistics
353 mutable std::mutex stats_mutex_;
354 std::atomic<size_t> collection_count_{0};
355 std::atomic<size_t> collection_errors_{0};
356
357private:
361 Derived& derived()
362 {
363 return static_cast<Derived&>(*this);
364 }
365
366 const Derived& derived() const
367 {
368 return static_cast<const Derived&>(*this);
369 }
370};

378template <typename T>
379concept QueryCollector = requires(T t, const collector_config& config,
380 const query_execution& exec,
381 const pool_stats& pool,
382 const cache_stats& cache,
383 const session_stats& session,
384 stats_map& stats) {
385 { T::collector_name } -> std::convertible_to<const char*>;
386 { t.do_initialize(config) } -> std::same_as<bool>;
387 { t.do_collect_query(exec) } -> std::same_as<void>;
388 { t.do_collect_pool(pool) } -> std::same_as<void>;
389 { t.do_collect_cache(cache) } -> std::same_as<void>;
390 { t.do_collect_session(session) } -> std::same_as<void>;
391 { t.is_available() } -> std::same_as<bool>;
392 { t.do_add_statistics(stats) } -> std::same_as<void>;
393 { t.do_get_metrics() } -> std::same_as<const query_server_metrics&>;
394};
