58// ============================================================================

61
62bool auth_token::is_expired() const noexcept
63{
64 if (expires_at == 0)
65 {
66 return false;
67 }
68
69 auto now = static_cast<uint64_t>(
70 std::chrono::duration_cast<std::chrono::milliseconds>(
71 std::chrono::system_clock::now().time_since_epoch())
72 .count());
73
74 return now > expires_at;
75}
76
77bool auth_token::is_valid() const noexcept
78{
79 return !token.empty() && !is_expired();
80}

102std::shared_ptr<container_module::value_container> query_request::serialize() const
103{
104#if KCENON_WITH_CONTAINER_SYSTEM
105 auto container = std::make_shared<container_module::value_container>();
106 container->set_message_type("query_request");
107
108 // Header
109 container->set_value("version", static_cast<int>(header.version));
110 container->set_value("message_id", static_cast<long long>(header.message_id));
111 container->set_value("timestamp", static_cast<long long>(header.timestamp));
112 container->set_value("correlation_id", header.correlation_id);
113
114 // Auth token
115 container->set_value("auth_token", token.token);
116 container->set_value("client_id", token.client_id);
117 container->set_value("token_expires", static_cast<long long>(token.expires_at));
118
119 // Query
120 container->set_value("query_type", static_cast<int>(type));
121 container->set_value("sql", sql);
122
123 // Options
124 container->set_value("timeout_ms", static_cast<int>(options.timeout_ms));
125 container->set_value("read_only", options.read_only);
126 container->set_value("isolation_level", options.isolation_level);
127 container->set_value("max_rows", static_cast<int>(options.max_rows));
128 container->set_value("include_metadata", options.include_metadata);
129
130 // Parameters count
131 container->set_value("params_count", static_cast<int>(params.size()));
132
133 // Serialize each parameter
134 for (size_t i = 0; i < params.size(); ++i)
135 {
136 const auto& param = params[i];
137 std::string prefix = "param_" + std::to_string(i) + "_";
138
139 container->set_value(prefix + "name", param.name);
140 container->set_value(prefix + "type", static_cast<int>(param.value.index()));
141
142 std::visit(
143 [&container, &prefix](auto&& arg)
144 {
145 using T = std::decay_t<decltype(arg)>;
146 if constexpr (std::is_same_v<T, std::monostate>)
147 {
148 // NULL value - nothing to store
149 }
150 else if constexpr (std::is_same_v<T, bool>)
151 {
152 container->set_value(prefix + "value_bool", arg);
153 }
154 else if constexpr (std::is_same_v<T, int64_t>)
155 {
156 container->set_value(prefix + "value_int", static_cast<long long>(arg));
157 }
158 else if constexpr (std::is_same_v<T, double>)
159 {
160 container->set_value(prefix + "value_double", arg);
161 }
162 else if constexpr (std::is_same_v<T, std::string>)
163 {
164 container->set_value(prefix + "value_string", arg);
165 }
166 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
167 {
168 container->set_value(prefix + "value_bytes", arg);
169 }
170 },
171 param.value);
172 }
173
174 return container;
175#else
176 return nullptr;
177#endif
178}

188

190

193 {
194 if (std::holds_alternative<int>(val->data))
195 {
196 request.header.version = static_cast<uint32_t>(std::get<int>(val->data));
197 }
198 }
199 if (auto val = container->get_value("message_id"))
200 {
201 if (std::holds_alternative<long long>(val->data))
202 {
203 request.header.message_id = static_cast<uint64_t>(std::get<long long>(val->data));
204 }
205 }

207 {
208 if (std::holds_alternative<long long>(val->data))
209 {
210 request.header.timestamp = static_cast<uint64_t>(std::get<long long>(val->data));
211 }
212 }
213 if (auto val = container->get_value("correlation_id"))
214 {
215 if (std::holds_alternative<std::string>(val->data))
216 {
217 request.header.correlation_id = std::get<std::string>(val->data);
218 }
219 }
220
221 // Auth token
222 if (auto val = container->get_value("auth_token"))
223 {
224 if (std::holds_alternative<std::string>(val->data))
225 {
226 request.token.token = std::get<std::string>(val->data);
227 }
228 }
229 if (auto val = container->get_value("client_id"))
230 {
231 if (std::holds_alternative<std::string>(val->data))
232 {
233 request.token.client_id = std::get<std::string>(val->data);
234 }
235 }
236 if (auto val = container->get_value("token_expires"))
237 {
238 if (std::holds_alternative<long long>(val->data))
239 {
240 request.token.expires_at = static_cast<uint64_t>(std::get<long long>(val->data));
241 }
242 }
243
244 // Query
245 if (auto val = container->get_value("query_type"))
246 {
247 if (std::holds_alternative<int>(val->data))
248 {
249 request.type = static_cast<query_type>(std::get<int>(val->data));
250 }
251 }
252 if (auto val = container->get_value("sql"))
253 {
254 if (std::holds_alternative<std::string>(val->data))
255 {
256 request.sql = std::get<std::string>(val->data);
257 }
258 }
259
260 // Options
261 if (auto val = container->get_value("timeout_ms"))
262 {
263 if (std::holds_alternative<int>(val->data))
264 {
265 request.options.timeout_ms = static_cast<uint32_t>(std::get<int>(val->data));
266 }
267 }
268 if (auto val = container->get_value("read_only"))
269 {
270 if (std::holds_alternative<bool>(val->data))
271 {
272 request.options.read_only = std::get<bool>(val->data);
273 }
274 }
275 if (auto val = container->get_value("isolation_level"))
276 {
277 if (std::holds_alternative<std::string>(val->data))
278 {
279 request.options.isolation_level = std::get<std::string>(val->data);
280 }
281 }
282 if (auto val = container->get_value("max_rows"))
283 {
284 if (std::holds_alternative<int>(val->data))
285 {
286 request.options.max_rows = static_cast<uint32_t>(std::get<int>(val->data));
287 }
288 }
289 if (auto val = container->get_value("include_metadata"))
290 {
291 if (std::holds_alternative<bool>(val->data))
292 {
293 request.options.include_metadata = std::get<bool>(val->data);
294 }
295 }
296
297 // Parameters
298 int params_count = 0;
299 if (auto val = container->get_value("params_count"))
300 {
301 if (std::holds_alternative<int>(val->data))
302 {
303 params_count = std::get<int>(val->data);
304 }
305 }
306
307 for (int i = 0; i < params_count; ++i)
308 {
309 std::string prefix = "param_" + std::to_string(i) + "_";
310 query_param param;
311
312 if (auto val = container->get_value(prefix + "name"))
313 {
314 if (std::holds_alternative<std::string>(val->data))
315 {
316 param.name = std::get<std::string>(val->data);
317 }
318 }
319
320 int param_type = 0;
321 if (auto val = container->get_value(prefix + "type"))
322 {
323 if (std::holds_alternative<int>(val->data))
324 {
325 param_type = std::get<int>(val->data);
326 }
327 }
328
329 switch (param_type)
330 {
331 case 0: // monostate (NULL)
332 param.value = std::monostate{};
333 break;
334 case 1: // bool
335 if (auto val = container->get_value(prefix + "value_bool"))
336 {
337 if (std::holds_alternative<bool>(val->data))
338 {
339 param.value = std::get<bool>(val->data);
340 }
341 }
342 break;
343 case 2: // int64_t
344 if (auto val = container->get_value(prefix + "value_int"))
345 {
346 if (std::holds_alternative<long long>(val->data))
347 {
348 param.value = static_cast<int64_t>(std::get<long long>(val->data));
349 }
350 }
351 break;
352 case 3: // double
353 if (auto val = container->get_value(prefix + "value_double"))
354 {
355 if (std::holds_alternative<double>(val->data))
356 {
357 param.value = std::get<double>(val->data);
358 }
359 }
360 break;
361 case 4: // string
362 if (auto val = container->get_value(prefix + "value_string"))
363 {
364 if (std::holds_alternative<std::string>(val->data))
365 {
366 param.value = std::get<std::string>(val->data);
367 }
368 }
369 break;
370 case 5: // bytes
371 if (auto val = container->get_value(prefix + "value_bytes"))
372 {
373 if (std::holds_alternative<std::vector<uint8_t>>(val->data))
374 {
375 param.value = std::get<std::vector<uint8_t>>(val->data);
376 }
377 }
378 break;
379 }
380
381 request.params.push_back(std::move(param));
382 }
383
384 return request;
385#else
386 return kcenon::common::error_info{
387 -2, "container_system not available", "query_protocol"
388 };
389#endif
390}

445std::shared_ptr<container_module::value_container> query_response::serialize() const
446{
447#if KCENON_WITH_CONTAINER_SYSTEM
448 auto container = std::make_shared<container_module::value_container>();
449 container->set_message_type("query_response");
450
451 // Header
452 container->set_value("version", static_cast<int>(header.version));
453 container->set_value("message_id", static_cast<long long>(header.message_id));
454 container->set_value("timestamp", static_cast<long long>(header.timestamp));
455 container->set_value("correlation_id", header.correlation_id);
456
457 // Status
458 container->set_value("status", static_cast<int>(status));
459 container->set_value("error_message", error_message);
460 container->set_value("affected_rows", static_cast<long long>(affected_rows));
461 container->set_value("execution_time_us", static_cast<long long>(execution_time_us));
462
463 // Column metadata
464 container->set_value("columns_count", static_cast<int>(columns.size()));
465 for (size_t i = 0; i < columns.size(); ++i)
466 {
467 const auto& col = columns[i];
468 std::string prefix = "col_" + std::to_string(i) + "_";
469
470 container->set_value(prefix + "name", col.name);
471 container->set_value(prefix + "type_name", col.type_name);
472 container->set_value(prefix + "type_id", static_cast<int>(col.type_id));
473 container->set_value(prefix + "nullable", col.nullable);
474 container->set_value(prefix + "precision", static_cast<int>(col.precision));
475 container->set_value(prefix + "scale", static_cast<int>(col.scale));
476 }
477
478 // Rows
479 container->set_value("rows_count", static_cast<int>(rows.size()));
480 for (size_t row_idx = 0; row_idx < rows.size(); ++row_idx)
481 {
482 const auto& row = rows[row_idx];
483 std::string row_prefix = "row_" + std::to_string(row_idx) + "_";
484
485 for (size_t cell_idx = 0; cell_idx < row.cells.size(); ++cell_idx)
486 {
487 std::string cell_prefix = row_prefix + "cell_" + std::to_string(cell_idx) + "_";
488
489 container->set_value(cell_prefix + "type",
490 static_cast<int>(row.cells[cell_idx].index()));
491
492 std::visit(
493 [&container, &cell_prefix](auto&& arg)
494 {
495 using T = std::decay_t<decltype(arg)>;
496 if constexpr (std::is_same_v<T, std::monostate>)
497 {
498 // NULL value
499 }
500 else if constexpr (std::is_same_v<T, bool>)
501 {
502 container->set_value(cell_prefix + "value_bool", arg);
503 }
504 else if constexpr (std::is_same_v<T, int64_t>)
505 {
506 container->set_value(cell_prefix + "value_int",
507 static_cast<long long>(arg));
508 }
509 else if constexpr (std::is_same_v<T, double>)
510 {
511 container->set_value(cell_prefix + "value_double", arg);
512 }
513 else if constexpr (std::is_same_v<T, std::string>)
514 {
515 container->set_value(cell_prefix + "value_string", arg);
516 }
517 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
518 {
519 container->set_value(cell_prefix + "value_bytes", arg);
520 }
521 },
522 row.cells[cell_idx]);
523 }
524 }
525
526 return container;
527#else
528 return nullptr;
529#endif
530}

540

637 }

645 }

651 }
652 }

669 {
670 rows_count = std::get<int>(val->data);
671 }
672 }
673

676 result_row row;
677 std::string row_prefix = "row_" + std::to_string(row_idx) + "_";
678
679 for (int cell_idx = 0; cell_idx < columns_count; ++cell_idx)
680 {
681 std::string cell_prefix = row_prefix + "cell_" + std::to_string(cell_idx) + "_";
682
683 int cell_type = 0;

685 {

690 }

701 if (std::holds_alternative<bool>(val->data))
702 {
703 cell = std::get<bool>(val->data);
704 }
705 }
706 break;
707 case 2: // int64_t
708 if (auto val = container->get_value(cell_prefix + "value_int"))
709 {
710 if (std::holds_alternative<long long>(val->data))
711 {
712 cell = static_cast<int64_t>(std::get<long long>(val->data));
713 }
714 }
715 break;
716 case 3: // double
717 if (auto val = container->get_value(cell_prefix + "value_double"))
718 {
719 if (std::holds_alternative<double>(val->data))
720 {
721 cell = std::get<double>(val->data);
722 }
723 }
724 break;
725 case 4: // string
726 if (auto val = container->get_value(cell_prefix + "value_string"))
727 {
728 if (std::holds_alternative<std::string>(val->data))
729 {
730 cell = std::get<std::string>(val->data);
731 }
732 }
733 break;
734 case 5: // bytes
735 if (auto val = container->get_value(cell_prefix + "value_bytes"))
736 {
737 if (std::holds_alternative<std::vector<uint8_t>>(val->data))
738 {
739 cell = std::get<std::vector<uint8_t>>(val->data);
740 }
741 }
742 break;
743 }
744
745 row.cells.push_back(std::move(cell));
746 }
747
748 response.rows.push_back(std::move(row));
749 }
750
751 return response;
752#else
753 return kcenon::common::error_info{
754 -2, "container_system not available", "query_protocol"
755 };
756#endif
