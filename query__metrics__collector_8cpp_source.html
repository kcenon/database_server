84bool query_metrics_collector::do_initialize(const collector_config& config)
85{
86 // Parse collector-specific configuration
87 if (auto it = config.find("track_query_types"); it != config.end())
88 {
89 options_.track_query_types = (it->second == "true" || it->second == "1");
90 }
91
92 if (auto it = config.find("track_latency_histogram"); it != config.end())
93 {
94 options_.track_latency_histogram = (it->second == "true" || it->second == "1");
95 }
96
97 if (auto it = config.find("histogram_buckets"); it != config.end())
98 {
99 try
100 {
101 options_.histogram_buckets = static_cast<uint32_t>(std::stoul(it->second));
102 }
103 catch (...)
104 {
105 // Keep default value on parse error
106 }
107 }
108
109 return true;
110}

112void query_metrics_collector::do_collect_query(const query_execution& exec)
113{
114 // Record basic query metrics (atomic operations, no lock needed)
115 metrics_.query_metrics.record_query(exec.latency_ns, exec.success, exec.timeout);
116
117 // Update query type counters if enabled
118 if (options_.track_query_types)
119 {
120 update_query_type_counter(exec.query_type);
121 }
122}

124void query_metrics_collector::do_collect_pool(const pool_stats& stats)
125{
126 // Update connection counts (atomic operations)
127 metrics_.pool_metrics.active_connections.store(
128 stats.active_connections, std::memory_order_relaxed);
129 metrics_.pool_metrics.idle_connections.store(
130 stats.idle_connections, std::memory_order_relaxed);
131 metrics_.pool_metrics.total_connections.store(
132 stats.total_connections, std::memory_order_relaxed);
133
134 // Record acquisition if applicable
135 if (stats.acquisition_time_ns > 0)
136 {
137 metrics_.pool_metrics.record_acquisition(
138 stats.acquisition_time_ns, stats.acquisition_success);
139 }
140
141 // Record pool exhaustion
142 if (stats.pool_exhausted)
143 {
144 metrics_.pool_metrics.record_exhaustion();
145 }
146}

148void query_metrics_collector::do_collect_cache(const cache_stats& stats)
149{
150 // Record hit or miss
151 if (stats.hit)
152 {
153 metrics_.cache_metrics.record_hit();
154 }
155 else
156 {
157 metrics_.cache_metrics.record_miss();
158 }
159
160 // Record eviction
161 if (stats.eviction)
162 {
163 metrics_.cache_metrics.record_eviction();
164 }
165
166 // Record expiration
167 if (stats.expiration)
168 {
169 metrics_.cache_metrics.record_expiration();
170 }
171
172 // Update cache size
173 metrics_.cache_metrics.cache_size_bytes.store(
174 stats.size_bytes, std::memory_order_relaxed);
175 metrics_.cache_metrics.cache_entries.store(
176 stats.entry_count, std::memory_order_relaxed);
177}

206void query_metrics_collector::do_add_statistics(stats_map& stats) const
207{
208 std::shared_lock<std::shared_mutex> lock(metrics_mutex_);
209
210 // Query metrics
211 stats["total_queries"] = static_cast<double>(
212 metrics_.query_metrics.total_queries.load(std::memory_order_relaxed));
213 stats["successful_queries"] = static_cast<double>(
214 metrics_.query_metrics.successful_queries.load(std::memory_order_relaxed));
215 stats["failed_queries"] = static_cast<double>(
216 metrics_.query_metrics.failed_queries.load(std::memory_order_relaxed));
217 stats["timeout_queries"] = static_cast<double>(
218 metrics_.query_metrics.timeout_queries.load(std::memory_order_relaxed));
219 stats["avg_query_latency_ms"] = metrics_.query_metrics.avg_query_latency_ms();
220 stats["query_success_rate"] = metrics_.query_metrics.success_rate();
221
222 // Cache metrics
223 stats["cache_hits"] = static_cast<double>(
224 metrics_.cache_metrics.cache_hits.load(std::memory_order_relaxed));
225 stats["cache_misses"] = static_cast<double>(
226 metrics_.cache_metrics.cache_misses.load(std::memory_order_relaxed));
227 stats["cache_hit_ratio"] = metrics_.cache_metrics.cache_hit_ratio();
228 stats["cache_evictions"] = static_cast<double>(
229 metrics_.cache_metrics.cache_evictions.load(std::memory_order_relaxed));
230
231 // Pool metrics
232 stats["active_connections"] = static_cast<double>(
233 metrics_.pool_metrics.active_connections.load(std::memory_order_relaxed));
234 stats["idle_connections"] = static_cast<double>(
235 metrics_.pool_metrics.idle_connections.load(std::memory_order_relaxed));
236 stats["pool_exhaustion_count"] = static_cast<double>(
237 metrics_.pool_metrics.pool_exhaustion_count.load(std::memory_order_relaxed));
238 stats["avg_acquisition_time_ms"] = metrics_.pool_metrics.avg_acquisition_time_ms();
239
240 // Session metrics
241 stats["active_sessions"] = static_cast<double>(
242 metrics_.session_metrics.active_sessions.load(std::memory_order_relaxed));
243 stats["total_sessions"] = static_cast<double>(
244 metrics_.session_metrics.total_sessions.load(std::memory_order_relaxed));
245 stats["avg_session_duration_sec"] = metrics_.session_metrics.avg_session_duration_sec();
246}

253void query_metrics_collector::update_query_type_counter(const std::string& query_type)
254{
255 // Convert to lowercase for comparison
256 std::string type_lower = query_type;
257 std::transform(type_lower.begin(), type_lower.end(), type_lower.begin(),
258 [](unsigned char c) { return std::tolower(c); });
259
260 if (type_lower == "select")
261 {
262 metrics_.query_metrics.select_queries.fetch_add(1, std::memory_order_relaxed);
263 }
264 else if (type_lower == "insert")
265 {
266 metrics_.query_metrics.insert_queries.fetch_add(1, std::memory_order_relaxed);
267 }
268 else if (type_lower == "update")
269 {
270 metrics_.query_metrics.update_queries.fetch_add(1, std::memory_order_relaxed);
271 }
272 else if (type_lower == "delete")
273 {
274 metrics_.query_metrics.delete_queries.fetch_add(1, std::memory_order_relaxed);
275 }
276 else
277 {
278 metrics_.query_metrics.other_queries.fetch_add(1, std::memory_order_relaxed);
279 }
280}
