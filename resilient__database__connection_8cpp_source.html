71kcenon::common::VoidResult resilient_database_connection::initialize(
72 const database::core::connection_config& config)
73{
74 std::lock_guard<std::mutex> lock(mutex_);
75
76 if (!backend_)
77 {
78 return kcenon::common::error_info{
79 -1,
80 "Backend is null",
81 "resilient_database_connection"
82 };
83 }
84
85 connection_config_ = config;
86 set_state(connection_state::connecting);
87
88 auto result = backend_->initialize(config);
89
90 if (result.is_err())
91 {
92 set_state(connection_state::failed);
93 last_error_message_ = result.error().message;
94 return result;
95 }
96
97 set_state(connection_state::connected);
98 reset_retry_state();
99
100 if (health_monitor_)
101 {
102 health_monitor_->start_monitoring();
103 }
104
105 return kcenon::common::ok();
106}

108kcenon::common::VoidResult resilient_database_connection::shutdown()
109{
110 stop_auto_recovery();
111
112 if (health_monitor_)
113 {
114 health_monitor_->stop_monitoring();
115 }
116
117 std::lock_guard<std::mutex> lock(mutex_);
118
119 if (!backend_)
120 {
121 return kcenon::common::ok();
122 }
123
124 auto result = backend_->shutdown();
125 set_state(connection_state::disconnected);
126
127 return result;
128}

171kcenon::common::VoidResult resilient_database_connection::begin_transaction()
172{
173 // Transactions require stable connection - don't retry during transaction
174 auto ensure_result = ensure_connected();
175 if (ensure_result.is_err())
176 {
177 return ensure_result;
178 }
179
180 return backend_->begin_transaction();

293kcenon::common::VoidResult resilient_database_connection::attempt_reconnect()
294{
295 if (!config_.enable_auto_reconnect)
296 {
297 return kcenon::common::error_info{
298 -2,
299 "Auto reconnect disabled",
300 "resilient_database_connection"
301 };
302 }
303
304 std::lock_guard<std::mutex> lock(mutex_);
305
306 if (retry_count_ >= config_.max_retries)
307 {
308 set_state(connection_state::failed);
309 last_error_message_ = "Max retries exceeded";
310 return kcenon::common::error_info{
311 -3,
312 "Max retries exceeded",
313 "resilient_database_connection"
314 };
315 }
316
317 set_state(connection_state::reconnecting);
318
319 // Calculate and apply backoff delay
320 auto delay = calculate_next_delay();
321 std::this_thread::sleep_for(delay);
322
323 // Attempt reconnection
324 if (!backend_)
325 {
326 set_state(connection_state::failed);
327 last_error_message_ = "Backend is null";
328 return kcenon::common::error_info{
329 -1,
330 "Backend is null",
331 "resilient_database_connection"
332 };
333 }
334
335 backend_->shutdown();
336
337 auto result = backend_->initialize(connection_config_);
338
339 if (result.is_err())
340 {
341 retry_count_++;
342 last_error_message_ = result.error().message;
343 set_state(connection_state::failed);
344 return result;
345 }
346
347 set_state(connection_state::connected);
348 reset_retry_state();
349
350 if (health_monitor_)
351 {
352 health_monitor_->reset_statistics();
353 health_monitor_->start_monitoring();
354 }
355
356 return kcenon::common::ok();
357}

359template <typename Func>
360auto resilient_database_connection::execute_with_retry(Func&& operation)
361 -> decltype(operation())
362{
363 using result_type = decltype(operation());
364
365 if (!backend_)
366 {
367 return kcenon::common::error_info{
368 -1,
369 "Backend is null",
370 "resilient_database_connection"
371 };
372 }
373
374 // Don't retry if in transaction
375 if (in_transaction())
376 {
377 return operation();
378 }
379
380 auto start_time = std::chrono::high_resolution_clock::now();
381
382 // Try operation
383 auto result = operation();
384
385 auto end_time = std::chrono::high_resolution_clock::now();
386 auto latency
387 = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
388
389 if (result.is_ok())
390 {
391 if (health_monitor_)
392 {
393 health_monitor_->record_success(latency);
394 }
395 return result;
396 }
397
398 // Operation failed - record and attempt reconnect if enabled
399 if (health_monitor_)
400 {
401 health_monitor_->record_failure(result.error().message);
402 }
403
404 if (!config_.enable_auto_reconnect)
405 {
406 std::lock_guard<std::mutex> lock(mutex_);
407 last_error_message_ = result.error().message;
408 return result;
409 }
410
411 // Attempt reconnection
412 auto reconnect_result = attempt_reconnect();
413 if (reconnect_result.is_err())
414 {
415 return result; // Return original error
416 }
417
418 // Retry operation after successful reconnection
419 auto retry_result = operation();
420
421 if (retry_result.is_ok() && health_monitor_)
422 {
423 health_monitor_->record_success(latency);
424 }
425
426 return retry_result;
427}

429std::chrono::milliseconds resilient_database_connection::calculate_next_delay()
430{
431 uint32_t attempts = retry_count_.load();
432 double delay_ms = config_.initial_delay.count()
433 * std::pow(config_.backoff_multiplier, attempts);
434
435 delay_ms = std::min(delay_ms, static_cast<double>(config_.max_delay.count()));
436
437 return std::chrono::milliseconds(static_cast<int64_t>(delay_ms));
438}
