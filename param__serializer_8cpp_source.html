53void serialize_params(std::shared_ptr<container_module::value_container>& container,
54 const std::vector<query_param>& params)
55{
56 container->set("params_count", static_cast<int>(params.size()));
57
58 for (size_t i = 0; i < params.size(); ++i)
59 {
60 const auto& param = params[i];
61 std::string prefix = "param_" + std::to_string(i) + "_";
62
63 container->set(prefix + "name", param.name);
64 serialize_variant_value(container, prefix, param.value);
65 }
66}

68std::vector<query_param> deserialize_params(
69 const std::shared_ptr<container_module::value_container>& container)
70{
71 std::vector<query_param> params;
72
73 int params_count = 0;
74 if (auto val = container->get("params_count"))
75 {
76 if (std::holds_alternative<int>(val->data))
77 {
78 params_count = std::get<int>(val->data);
79 }
80 }
81
82 for (int i = 0; i < params_count; ++i)
83 {
84 std::string prefix = "param_" + std::to_string(i) + "_";
85 query_param param;
86
87 if (auto val = container->get(prefix + "name"))
88 {
89 if (std::holds_alternative<std::string>(val->data))
90 {
91 param.name = std::get<std::string>(val->data);
92 }
93 }
94
95 param.value = deserialize_variant_value<query_param::param_value>(container, prefix);
96 params.push_back(std::move(param));
97 }
98
99 return params;
100}
