46kcenon::common::Result<query_response> query_cache::get(const std::string& cache_key)
47{
48 if (!config_.enabled)
49 {
50 return kcenon::common::error_info{
51 kcenon::common::error_codes::NOT_INITIALIZED,
52 "Cache is disabled",
53 "query_cache"};
54 }
55
56 std::unique_lock lock(mutex_);
57
58 auto it = cache_map_.find(cache_key);
59 if (it == cache_map_.end())
60 {
61 ++metrics_.misses;
62 return kcenon::common::error_info{
63 kcenon::common::error_codes::NOT_FOUND,
64 "Cache miss: key not found",
65 "query_cache"};
66 }
67
68 auto& entry = *(it->second);
69
70 if (is_expired(entry))
71 {
72 ++metrics_.expirations;
73 ++metrics_.misses;
74 remove_entry(it->second);
75 return kcenon::common::error_info{
76 kcenon::common::error_codes::NOT_FOUND,
77 "Cache entry expired",
78 "query_cache"};
79 }
80
81 ++metrics_.hits;
82
83 if (config_.enable_lru)
84 {
85 lru_list_.splice(lru_list_.begin(), lru_list_, it->second);
86 }
87
88 return kcenon::common::ok(entry.response);
89}

91kcenon::common::VoidResult query_cache::put(
92 const std::string& cache_key,
93 const query_response& response,
94 const std::unordered_set<std::string>& table_names)
95{
96 if (!config_.enabled)
97 {
98 return kcenon::common::error_info{
99 kcenon::common::error_codes::NOT_INITIALIZED,
100 "Cache is disabled",
101 "query_cache"};
102 }
103
104 if (response.status != status_code::ok)
105 {
106 return kcenon::common::error_info{
107 kcenon::common::error_codes::INVALID_ARGUMENT,
108 "Cannot cache error response",
109 "query_cache"};
110 }
111
112 auto estimated_size = estimate_size(response);
113 if (estimated_size > config_.max_result_size_bytes)
114 {
115 ++metrics_.skipped_too_large;
116 return kcenon::common::error_info{
117 kcenon::common::error_codes::INVALID_ARGUMENT,
118 "Cache entry too large",
119 "query_cache"};
120 }
121
122 std::unique_lock lock(mutex_);
123
124 auto existing = cache_map_.find(cache_key);
125 if (existing != cache_map_.end())
126 {
127 remove_entry(existing->second);
128 }
129
130 while (cache_map_.size() >= config_.max_entries && !lru_list_.empty())
131 {
132 evict_lru();
133 }
134
135 cache_entry entry;
136 entry.key = cache_key;
137 entry.response = response;
138 entry.tables = table_names;
139 entry.estimated_size = estimated_size;
140
141 if (config_.ttl_seconds > 0)
142 {
143 entry.expires_at = std::chrono::steady_clock::now()
144 + std::chrono::seconds(config_.ttl_seconds);
145 }
146 else
147 {
148 entry.expires_at = std::chrono::steady_clock::time_point::max();
149 }
150
151 lru_list_.push_front(std::move(entry));
152 cache_map_[cache_key] = lru_list_.begin();
153
154 for (const auto& table : table_names)
155 {
156 table_map_[table].insert(cache_key);
157 }
158
159 ++metrics_.puts;
160 return kcenon::common::ok();
161}

163std::string query_cache::make_key(const query_request& request)
164{
165 std::hash<std::string> hasher;
166 size_t hash_value = hasher(request.sql);
167
168 for (const auto& param : request.params)
169 {
170 hash_value ^= hasher(param.name) + 0x9e3779b9 + (hash_value << 6) + (hash_value >> 2);
171
172 std::visit(
173 [&hash_value, &hasher](auto&& arg) {
174 using T = std::decay_t<decltype(arg)>;
175 if constexpr (std::is_same_v<T, std::monostate>)
176 {
177 hash_value ^= 0 + 0x9e3779b9 + (hash_value << 6) + (hash_value >> 2);
178 }
179 else if constexpr (std::is_same_v<T, bool>)
180 {
181 hash_value ^= std::hash<bool>{}(arg) + 0x9e3779b9 + (hash_value << 6)
182 + (hash_value >> 2);
183 }
184 else if constexpr (std::is_same_v<T, int64_t>)
185 {
186 hash_value ^= std::hash<int64_t>{}(arg) + 0x9e3779b9 + (hash_value << 6)
187 + (hash_value >> 2);
188 }
189 else if constexpr (std::is_same_v<T, double>)
190 {
191 hash_value ^= std::hash<double>{}(arg) + 0x9e3779b9 + (hash_value << 6)
192 + (hash_value >> 2);
193 }
194 else if constexpr (std::is_same_v<T, std::string>)
195 {
196 hash_value ^= hasher(arg) + 0x9e3779b9 + (hash_value << 6) + (hash_value >> 2);
197 }
198 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
199 {
200 for (auto byte : arg)
201 {
202 hash_value ^= std::hash<uint8_t>{}(byte) + 0x9e3779b9 + (hash_value << 6)
203 + (hash_value >> 2);
204 }
205 }
206 },
207 param.value);
208 }
209
210 hash_value ^= std::hash<uint32_t>{}(request.options.max_rows) + 0x9e3779b9 + (hash_value << 6)
211 + (hash_value >> 2);
212
213 std::ostringstream oss;
214 oss << std::hex << hash_value;
215 return oss.str();
216}

218kcenon::common::VoidResult query_cache::invalidate(const std::string& table_name)
219{
220 std::unique_lock lock(mutex_);
221
222 auto it = table_map_.find(table_name);
223 if (it == table_map_.end())
224 {
225 return kcenon::common::ok(); // No entries to invalidate is not an error
226 }
227
228 // Copy keys to remove since remove_entry modifies table_map_
229 auto keys_to_remove = it->second;
230
231 // Clear the table mapping first to avoid iterator invalidation
232 table_map_.erase(it);
233
234 for (const auto& key : keys_to_remove)
235 {
236 auto cache_it = cache_map_.find(key);
237 if (cache_it != cache_map_.end())
238 {
239 remove_entry(cache_it->second);
240 ++metrics_.invalidations;
241 }
242 }
243
244 return kcenon::common::ok();
245}

305void query_cache::evict_lru()
306{
307 if (lru_list_.empty())
308 {
309 return;
310 }
311
312 auto& entry = lru_list_.back();
313 remove_entry(std::prev(lru_list_.end()));
314 ++metrics_.evictions;

317size_t query_cache::estimate_size(const query_response& response)
318{
319 size_t size = sizeof(query_response);
320
321 size += response.error_message.size();
322
323 for (const auto& col : response.columns)
324 {
325 size += sizeof(column_metadata);
326 size += col.name.size();
327 size += col.type_name.size();
328 }
329
330 for (const auto& row : response.rows)
331 {
332 size += sizeof(result_row);
333 for (const auto& cell : row.cells)
334 {
335 size += sizeof(result_row::cell_value);
336 std::visit(
337 [&size](auto&& arg) {
338 using T = std::decay_t<decltype(arg)>;
339 if constexpr (std::is_same_v<T, std::string>)
340 {
341 size += arg.size();
342 }
343 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
344 {
345 size += arg.size();
346 }
347 },
348 cell);
349 }
350 }
351
352 return size;
353}

355void query_cache::remove_entry(cache_list::iterator it)
356{
357 const auto& entry = *it;
358
359 for (const auto& table : entry.tables)
360 {
361 auto table_it = table_map_.find(table);
362 if (table_it != table_map_.end())
363 {
364 table_it->second.erase(entry.key);
365 if (table_it->second.empty())
366 {
367 table_map_.erase(table_it);
368 }
369 }
370 }
371
372 cache_map_.erase(entry.key);
373 lru_list_.erase(it);
374}
