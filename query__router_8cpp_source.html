46uint64_t current_timestamp_us()
47{
48 return static_cast<uint64_t>(
49 std::chrono::duration_cast<std::chrono::microseconds>(
50 std::chrono::steady_clock::now().time_since_epoch())
51 .count());
52}

54pooling::connection_priority get_priority_for_query_type(query_type type)
55{
56 switch (type)
57 {
58 case query_type::select:
59 return pooling::PRIORITY_NORMAL_QUERY;
60 case query_type::insert:
61 case query_type::update:
62 case query_type::del:
63 return pooling::PRIORITY_TRANSACTION;
64 case query_type::execute:
65 return pooling::PRIORITY_TRANSACTION;
66 case query_type::batch:
67 return pooling::PRIORITY_TRANSACTION;
68 case query_type::ping:
69 return pooling::PRIORITY_HEALTH_CHECK;
70 default:
71 return pooling::PRIORITY_NORMAL_QUERY;
72 }
73}

92query_response query_router::execute(const query_request& request)
93{
94 auto start_time = current_timestamp_us();
95
96 // Check if router is ready
97 if (!is_ready())
98 {
99 record_metrics(false, false, 0);
100 return query_response(request.header.message_id, status_code::no_connection,
101 "No connection pool available");
102 }
103
104 // Check concurrent query limit
105 auto current = active_queries_.fetch_add(1, std::memory_order_relaxed);
106 if (current >= config_.max_concurrent_queries)
107 {
108 active_queries_.fetch_sub(1, std::memory_order_relaxed);
109 record_metrics(false, false, 0);
110 return query_response(request.header.message_id, status_code::server_busy,
111 "Maximum concurrent queries exceeded");
112 }
113
114 // Execute based on query type
115 query_response response(request.header.message_id);
116
117 try
118 {
119 switch (request.type)
120 {
121 case query_type::select:
122 response = execute_select(request);
123 break;
124
125 case query_type::insert:
126 case query_type::update:
127 case query_type::del:
128 response = execute_modify(request);
129 break;
130
131 case query_type::execute:
132 response = execute_procedure(request);
133 break;
134
135 case query_type::batch:
136 // Batch queries not yet implemented
137 response = query_response(request.header.message_id, status_code::error,
138 "Batch queries not yet implemented");
139 break;
140
141 case query_type::ping:
142 // Ping is handled at gateway level
143 response = query_response(request.header.message_id);
144 break;
145
146 default:
147 response = query_response(request.header.message_id, status_code::invalid_query,
148 "Unknown query type");
149 break;
150 }
151 }
152 catch (const std::exception& e)
153 {
154 response = query_response(request.header.message_id, status_code::error,
155 std::string("Exception: ") + e.what());
156 }
157
158 active_queries_.fetch_sub(1, std::memory_order_relaxed);
159
160 // Calculate execution time
161 auto end_time = current_timestamp_us();
162 auto execution_time = end_time - start_time;
163 response.execution_time_us = execution_time;
164
165 // Record metrics
166 bool is_success = response.is_success();
167 bool is_timeout = response.status == status_code::timeout;
168 record_metrics(is_success, is_timeout, execution_time);
169
170 return response;
171}

214query_response query_router::execute_select(const query_request& request)
215{
216 std::shared_ptr<pooling::connection_pool> pool;
217 {
218 std::lock_guard<std::mutex> lock(pool_mutex_);
219 pool = pool_;
220 }
221
222 if (!pool)
223 {
224 return query_response(request.header.message_id, status_code::no_connection,
225 "Connection pool not available");
226 }
227
228 // Get priority based on query type
229 auto priority = get_priority_for_query_type(request.type);
230
231 // Acquire connection with timeout
232 auto timeout_ms = request.options.timeout_ms > 0
233 ? request.options.timeout_ms
234 : config_.default_timeout_ms;
235
236 auto future = pool->acquire_connection(priority);
237
238 auto status = future.wait_for(std::chrono::milliseconds(timeout_ms));
239 if (status == std::future_status::timeout)
240 {
241 return query_response(request.header.message_id, status_code::timeout,
242 "Connection acquisition timeout");
243 }
244
245 auto conn_result = future.get();
246 if (!conn_result.is_ok())
247 {
248 return query_response(request.header.message_id, status_code::no_connection,
249 "Failed to acquire connection: " +
250 conn_result.error().message);
251 }
252
253 auto connection = conn_result.value();
254
255 // Execute query
256 try
257 {
258 auto db = connection->get();
259 if (!db)
260 {
261 pool->release_connection(connection);
262 return query_response(request.header.message_id, status_code::error,
263 "Invalid database connection");
264 }
265
266 // Use select_query for SELECT statements
267 auto db_result = db->select_query(request.sql);
268
269 query_response response(request.header.message_id);
270
271 if (!db_result.empty())
272 {
273 // Extract column metadata from first row's keys
274 if (!db_result.empty())
275 {
276 for (const auto& [col_name, value] : db_result.front())
277 {
278 column_metadata meta;
279 meta.name = col_name;
280 // Determine type name from variant
281 std::visit(
282 [&meta](const auto& val)
283 {
284 using T = std::decay_t<decltype(val)>;
285 if constexpr (std::is_same_v<T, std::string>)
286 meta.type_name = "string";
287 else if constexpr (std::is_same_v<T, int64_t>)
288 meta.type_name = "integer";
289 else if constexpr (std::is_same_v<T, double>)
290 meta.type_name = "double";
291 else if constexpr (std::is_same_v<T, bool>)
292 meta.type_name = "boolean";
293 else
294 meta.type_name = "null";
295 },
296 value);
297 response.columns.push_back(std::move(meta));
298 }
299 }
300
301 // Convert rows
302 for (const auto& db_row : db_result)
303 {
304 result_row row;
305 for (const auto& [col_name, cell] : db_row)
306 {
307 std::visit(
308 [&row](const auto& val)
309 {
310 using T = std::decay_t<decltype(val)>;
311 if constexpr (std::is_same_v<T, std::nullptr_t>)
312 row.cells.push_back(std::monostate{});
313 else if constexpr (std::is_same_v<T, int64_t>)
314 row.cells.push_back(val);
315 else if constexpr (std::is_same_v<T, double>)
316 row.cells.push_back(val);
317 else if constexpr (std::is_same_v<T, bool>)
318 row.cells.push_back(val);
319 else if constexpr (std::is_same_v<T, std::string>)
320 row.cells.push_back(val);
321 else
322 row.cells.push_back(std::monostate{});
323 },
324 cell);
325 }
326 response.rows.push_back(std::move(row));
327 }
328 }
329
330 pool->release_connection(connection);
331 return response;
332 }
333 catch (const std::exception& e)
334 {
335 pool->release_connection(connection);
336 return query_response(request.header.message_id, status_code::error,
337 std::string("Query execution error: ") + e.what());
338 }

341query_response query_router::execute_modify(const query_request& request)
342{
343 std::shared_ptr<pooling::connection_pool> pool;
344 {
345 std::lock_guard<std::mutex> lock(pool_mutex_);
346 pool = pool_;
347 }
348
349 if (!pool)
350 {
351 return query_response(request.header.message_id, status_code::no_connection,
352 "Connection pool not available");
353 }
354
355 auto priority = get_priority_for_query_type(request.type);
356 auto timeout_ms = request.options.timeout_ms > 0
357 ? request.options.timeout_ms
358 : config_.default_timeout_ms;
359
360 auto future = pool->acquire_connection(priority);
361
362 auto status = future.wait_for(std::chrono::milliseconds(timeout_ms));
363 if (status == std::future_status::timeout)
364 {
365 return query_response(request.header.message_id, status_code::timeout,
366 "Connection acquisition timeout");
367 }
368
369 auto conn_result = future.get();
370 if (!conn_result.is_ok())
371 {
372 return query_response(request.header.message_id, status_code::no_connection,
373 "Failed to acquire connection: " +
374 conn_result.error().message);
375 }
376
377 auto connection = conn_result.value();
378
379 try
380 {
381 auto db = connection->get();
382 if (!db)
383 {
384 pool->release_connection(connection);
385 return query_response(request.header.message_id, status_code::error,
386 "Invalid database connection");
387 }
388
389 query_response response(request.header.message_id);
390 unsigned int affected_rows = 0;
391
392 // Use appropriate method based on query type
393 switch (request.type)
394 {
395 case query_type::insert:
396 affected_rows = db->insert_query(request.sql);
397 break;
398 case query_type::update:
399 affected_rows = db->update_query(request.sql);
400 break;
401 case query_type::del:
402 affected_rows = db->delete_query(request.sql);
403 break;
404 default:
405 // Generic execute for other types
406 if (!db->execute_query(request.sql))
407 {
408 pool->release_connection(connection);
409 return query_response(request.header.message_id, status_code::error,
410 "Query execution failed");
411 }
412 break;
413 }
414
415 response.affected_rows = static_cast<uint64_t>(affected_rows);
416
417 pool->release_connection(connection);
418 return response;
419 }
420 catch (const std::exception& e)
421 {
422 pool->release_connection(connection);
423 return query_response(request.header.message_id, status_code::error,
424 std::string("Query execution error: ") + e.what());
425 }

435void query_router::record_metrics(bool success, bool timeout, uint64_t execution_time_us)
436{
437 if (!config_.enable_metrics)
438 {
439 return;
440 }
441
442 metrics_.total_queries.fetch_add(1, std::memory_order_relaxed);
443
444 if (success)
445 {
446 metrics_.successful_queries.fetch_add(1, std::memory_order_relaxed);
447 }
448 else
449 {
450 metrics_.failed_queries.fetch_add(1, std::memory_order_relaxed);
451 }
452
453 if (timeout)
454 {
455 metrics_.timeout_queries.fetch_add(1, std::memory_order_relaxed);
456 }
457
458 metrics_.total_execution_time_us.fetch_add(execution_time_us, std::memory_order_relaxed);
459}
