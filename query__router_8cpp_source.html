95i_query_handler* query_router::find_handler(query_type type) const
96{
97 // First check custom handlers
98 {
99 std::lock_guard<std::mutex> lock(handlers_mutex_);
100 for (const auto& handler : handlers_)
101 {
102 if (handler->can_handle(type))
103 {
104 return handler.get();
105 }
106 }
107 }
108
109 // Return nullptr - built-in handlers are used directly in execute_with_handler
110 return nullptr;
111}

113query_response query_router::execute_with_handler(const query_request& request)
114{
115 auto ctx = get_handler_context();
116
117 // Check for custom handler first
118 auto* custom_handler = find_handler(request.type);
119 if (custom_handler != nullptr)
120 {
121 return custom_handler->handle(request, ctx);
122 }
123
124 // Use built-in CRTP handlers (zero virtual dispatch overhead)
125 switch (request.type)
126 {
127 case query_type::select:
128 return select_handler_.handle(request, ctx);
129
130 case query_type::insert:
131 return insert_handler_.handle(request, ctx);
132
133 case query_type::update:
134 return update_handler_.handle(request, ctx);
135
136 case query_type::del:
137 return delete_handler_.handle(request, ctx);
138
139 case query_type::execute:
140 return execute_handler_.handle(request, ctx);
141
142 case query_type::ping:
143 return ping_handler_.handle(request, ctx);
144
145 case query_type::batch:
146 return batch_handler_.handle(request, ctx);
147
148 default:
149 return query_response(request.header.message_id, status_code::invalid_query,
150 "Unknown query type");
151 }
152}

160kcenon::common::Result<query_response> query_router::execute(const query_request& request)
161{
162 auto start_time = current_timestamp_us();
163
164 // Check if router is ready
165 if (!is_ready())
166 {
167 record_metrics(false, false, 0);
168 return kcenon::common::error_info{
169 kcenon::common::error_codes::NOT_INITIALIZED,
170 "No connection pool available",
171 "query_router"};
172 }
173
174 // Check concurrent query limit
175 auto current = active_queries_.fetch_add(1, std::memory_order_relaxed);
176 if (current >= config_.max_concurrent_queries)
177 {
178 active_queries_.fetch_sub(1, std::memory_order_relaxed);
179 record_metrics(false, false, 0);
180 return kcenon::common::error_info{
181 kcenon::common::error_codes::INTERNAL_ERROR,
182 "Maximum concurrent queries exceeded",
183 "query_router"};
184 }
185
186 // Execute using CRTP handlers
187 query_response response(request.header.message_id);
188
189 try
190 {
191 response = execute_with_handler(request);
192 }
193 catch (const std::exception& e)
194 {
195 active_queries_.fetch_sub(1, std::memory_order_relaxed);
196 record_metrics(false, false, 0);
197 return kcenon::common::error_info{
198 kcenon::common::error_codes::INTERNAL_ERROR,
199 std::string("Exception: ") + e.what(),
200 "query_router"};
201 }
202
203 active_queries_.fetch_sub(1, std::memory_order_relaxed);
204
205 // Calculate execution time
206 auto end_time = current_timestamp_us();
207 auto execution_time = end_time - start_time;
208 response.execution_time_us = execution_time;
209
210 // Record metrics
211 bool is_success = response.is_success();
212 bool is_timeout = response.status == status_code::timeout;
213 record_metrics(is_success, is_timeout, execution_time);
214
215 return kcenon::common::ok(std::move(response));
216}

221class async_query_job : public kcenon::common::interfaces::IJob
222{
223public:
224 async_query_job(query_router* router,
225 query_request request,
226 std::function<void(query_response)> callback)
227 : router_(router)
228 , request_(std::move(request))
229 , callback_(std::move(callback))
230 {
231 }
232
233 kcenon::common::VoidResult execute() override
234 {
235 if (router_)
236 {
237 auto result = router_->execute(request_);
238 if (callback_)
239 {
240 if (result.is_ok())
241 {
242 callback_(std::move(result.value()));
243 }
244 else
245 {
246 // Create error response for callback
247 query_response error_response(
248 request_.header.message_id,
249 status_code::error,
250 result.error().message);
251 callback_(std::move(error_response));
252 }
253 }
254 }
255 return kcenon::common::ok();
256 }
257
258 std::string get_name() const override { return "async_query"; }
259 int get_priority() const override { return 0; }
260
261private:
262 query_router* router_;
263 query_request request_;
264 std::function<void(query_response)> callback_;
265};

267void query_router::execute_async(const query_request& request,
268 std::function<void(query_response)> callback)
269{
270 std::shared_ptr<kcenon::common::interfaces::IExecutor> exec;
271 {
272 std::lock_guard<std::mutex> lock(executor_mutex_);
273 exec = executor_;
274 }
275
276 if (exec)
277 {
278 // Use IExecutor for async execution
279 auto job = std::make_unique<async_query_job>(this, request, std::move(callback));
280 auto result = exec->execute(std::move(job));
281 // Fire and forget - the callback will be invoked when done
282 (void)result;
283 }
284 else
285 {
286 // Fallback to std::async if no executor provided
287 std::async(std::launch::async,
288 [this, request, callback = std::move(callback)]()
289 {
290 auto result = execute(request);
291 if (callback)
292 {
293 if (result.is_ok())
294 {
295 callback(std::move(result.value()));
296 }
297 else
298 {
299 query_response error_response(
300 request.header.message_id,
301 status_code::error,
302 result.error().message);
303 callback(std::move(error_response));
304 }
305 }
306 });
307 }

368void query_router::record_metrics(bool success, bool timeout, uint64_t execution_time_us)
369{
370 if (!config_.enable_metrics)
371 {
372 return;
373 }
374
375 metrics_.total_queries.fetch_add(1, std::memory_order_relaxed);
376
377 if (success)
378 {
379 metrics_.successful_queries.fetch_add(1, std::memory_order_relaxed);
380 }
381 else
382 {
383 metrics_.failed_queries.fetch_add(1, std::memory_order_relaxed);
384 }
385
386 if (timeout)
387 {
388 metrics_.timeout_queries.fetch_add(1, std::memory_order_relaxed);
389 }
390
391 metrics_.total_execution_time_us.fetch_add(execution_time_us, std::memory_order_relaxed);
392}
