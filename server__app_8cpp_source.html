77kcenon::common::VoidResult server_app::initialize(const server_config& config)
78{
79 if (state_ != server_state::uninitialized)
80 {
81 std::cerr << "Server already initialized" << std::endl;
82 return kcenon::common::error_info{
83 kcenon::common::error_codes::ALREADY_EXISTS,
84 "Server already initialized",
85 "server_app"};
86 }
87
88 config_ = config;
89
90 if (!config_.validate())
91 {
92 std::string error_msg = "Configuration validation failed:";
93 for (const auto& error : config_.validation_errors())
94 {
95 std::cerr << " - " << error << std::endl;
96 error_msg += " " + error + ";";
97 }
98 return kcenon::common::error_info{
99 kcenon::common::error_codes::INVALID_ARGUMENT,
100 error_msg,
101 "server_app"};
102 }
103
104 if (!do_initialize())
105 {
106 return kcenon::common::error_info{
107 kcenon::common::error_codes::INTERNAL_ERROR,
108 "Server initialization failed",
109 "server_app"};
110 }
111
112 state_ = server_state::initialized;
113 return kcenon::common::ok();
114}

116bool server_app::do_initialize()
117{
118 // Setup signal handlers
119 setup_signal_handlers();
120
121 // Note: IExecutor integration is available but optional.
122 // Modules will use std::async fallback when executor is not provided.
123 // To enable IExecutor, inject it via set_executor() methods on query_router
124 // and resilience components.
125
126 // Initialize query router
127 gateway::router_config router_cfg;
128 router_cfg.default_timeout_ms = config_.network.connection_timeout_ms;
129 router_cfg.max_concurrent_queries = config_.network.max_connections;
130 router_cfg.enable_metrics = true;
131
132 query_router_ = std::make_unique<gateway::query_router>(router_cfg);
133
134 std::cout << "Query router initialized" << std::endl;
135
136 // Initialize gateway server
137 gateway::gateway_config gw_config;
138 gw_config.server_id = config_.name;
139 gw_config.port = config_.network.port;
140 gw_config.max_connections = config_.network.max_connections;
141 gw_config.idle_timeout_ms = config_.network.connection_timeout_ms;
142
143 gateway_ = std::make_unique<gateway::gateway_server>(gw_config);
144
145 // Set up connection callbacks for logging
146 gateway_->set_connection_callback(
147 [this](const gateway::client_session& session)
148 {
149 std::cout << "Client connected: " << session.session_id << std::endl;
150 });
151
152 gateway_->set_disconnection_callback(
153 [](const std::string& session_id)
154 {
155 std::cout << "Client disconnected: " << session_id << std::endl;
156 });
157
158 // Set up request handler to route queries through query_router
159 gateway_->set_request_handler(
160 [this](const gateway::client_session& session, const gateway::query_request& request)
161 -> gateway::query_response
162 {
163 (void)session; // Session info available for future enhancements (logging, rate limiting)
164
165 // Execute query through query router
166 auto result = query_router_->execute(request);
167 if (result.is_ok())
168 {
169 return std::move(result.value());
170 }
171 else
172 {
173 return gateway::query_response(
174 request.header.message_id,
175 gateway::status_code::error,
176 result.error().message);
177 }
178 });
179
180 std::cout << "Server '" << config_.name << "' initialized" << std::endl;
181 std::cout << " Listen address: " << config_.network.host << ":" << config_.network.port
182 << std::endl;
183
184 return true;
185}

187int server_app::run()
188{
189 if (state_ != server_state::initialized)
190 {
191 std::cerr << "Server not initialized" << std::endl;
192 return 1;
193 }
194
195 state_ = server_state::starting;
196
197 // Start gateway server
198 if (gateway_)
199 {
200 auto result = gateway_->start();
201 if (result.is_err())
202 {
203 std::cerr << "Failed to start gateway server: " << result.error().message
204 << std::endl;
205 state_ = server_state::stopped;
206 return 1;
207 }
208 }
209
210 // Start connection pool health monitoring if pool is configured
211 if (connection_pool_)
212 {
213 connection_pool_->schedule_health_check();
214 std::cout << "Connection pool health monitoring started" << std::endl;
215 }
216
217 state_ = server_state::running;
218 std::cout << "Server is running. Press Ctrl+C to stop." << std::endl;
219
220 // Main event loop - wait for shutdown signal
221 while (state_ == server_state::running)
222 {
223 std::this_thread::sleep_for(std::chrono::milliseconds(100));
224 }
225
226 state_ = server_state::stopped;
227 std::cout << "Server stopped" << std::endl;
228
229 return 0;
230}

232void server_app::stop()
233{
234 if (state_ != server_state::running)
235 {
236 return;
237 }
238
239 state_ = server_state::stopping;
240
241 // Stop gateway server
242 if (gateway_)
243 {
244 auto result = gateway_->stop();
245 if (result.is_err())
246 {
247 std::cerr << "Warning: Failed to stop gateway server: " << result.error().message
248 << std::endl;
249 }
250 }
251
252 // Shutdown connection pool
253 if (connection_pool_)
254 {
255 connection_pool_->request_shutdown();
256 connection_pool_->shutdown();
257 }
258
259 std::cout << "Shutdown requested" << std::endl;
260}

262void server_app::do_cleanup()
263{
264 // Cleanup query router
265 query_router_.reset();
266
267 // Cleanup connection pool
268 connection_pool_.reset();
269
270 // Cleanup gateway server
271 gateway_.reset();
272
273 // Shutdown executor if set
274 if (executor_)
275 {
276 executor_->shutdown(true);
277 executor_.reset();
278 }
279
280 if (instance_ == this)
281 {
282 instance_ = nullptr;
283 }
284}

318void server_app::setup_signal_handlers()
319{
320 instance_ = this;
321
322#ifdef _WIN32
323 std::signal(SIGINT, signal_handler);
324 std::signal(SIGTERM, signal_handler);
325#else
326 struct sigaction sa;
327 sa.sa_handler = signal_handler;
328 sigemptyset(&sa.sa_mask);
329 sa.sa_flags = 0;
330
331 sigaction(SIGINT, &sa, nullptr);
332 sigaction(SIGTERM, &sa, nullptr);
333#endif
334}
