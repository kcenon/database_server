63kcenon::common::VoidResult server_app::initialize(const std::string& config_path)
64{
65 // Create default logger if not set
66 if (!logger_)
67 {
68 logger_ = logging::create_console_logger();
69 }
70
71 auto loaded_config = server_config::load_from_file(config_path);
72 if (!loaded_config)
73 {
74 logger_->log(kcenon::common::interfaces::log_level::error,
75 "Failed to load configuration from: " + config_path);
76 return kcenon::common::error_info{
77 kcenon::common::error_codes::INVALID_ARGUMENT,
78 "Failed to load configuration from: " + config_path,
79 "server_app"};
80 }
81
82 return initialize(*loaded_config);
83}

85kcenon::common::VoidResult server_app::initialize(const server_config& config)
86{
87 // Create default logger if not set
88 if (!logger_)
89 {
90 logger_ = logging::create_console_logger();
91 }
92
93 if (state_ != server_state::uninitialized)
94 {
95 logger_->log(kcenon::common::interfaces::log_level::error,
96 std::string("Server already initialized"));
97 return kcenon::common::error_info{
98 kcenon::common::error_codes::ALREADY_EXISTS,
99 "Server already initialized",
100 "server_app"};
101 }
102
103 config_ = config;
104
105 if (!config_.validate())
106 {
107 std::ostringstream error_oss;
108 error_oss << "Configuration validation failed:";
109 for (const auto& error : config_.validation_errors())
110 {
111 logger_->log(kcenon::common::interfaces::log_level::error,
112 " - " + error);
113 error_oss << " " << error << ";";
114 }
115 return kcenon::common::error_info{
116 kcenon::common::error_codes::INVALID_ARGUMENT,
117 error_oss.str(),
118 "server_app"};
119 }
120
121 if (!do_initialize())
122 {
123 return kcenon::common::error_info{
124 kcenon::common::error_codes::INTERNAL_ERROR,
125 "Server initialization failed",
126 "server_app"};
127 }
128
129 state_ = server_state::initialized;
130 return kcenon::common::ok();
131}

133bool server_app::do_initialize()
134{
135 // Setup signal handlers
136 setup_signal_handlers();
137
138 // Note: IExecutor integration is available but optional.
139 // Modules will use std::async fallback when executor is not provided.
140 // To enable IExecutor, inject it via set_executor() methods on query_router
141 // and resilience components.
142
143 // Initialize query router
144 gateway::router_config router_cfg;
145 router_cfg.default_timeout_ms = config_.network.connection_timeout_ms;
146 router_cfg.max_concurrent_queries = config_.network.max_connections;
147 router_cfg.enable_metrics = true;
148
149 query_router_ = std::make_unique<gateway::query_router>(router_cfg);
150
151 logger_->log(kcenon::common::interfaces::log_level::info,
152 std::string("Query router initialized"));
153
154 // Initialize gateway server
155 gateway::gateway_config gw_config;
156 gw_config.server_id = config_.name;
157 gw_config.port = config_.network.port;
158 gw_config.max_connections = config_.network.max_connections;
159 gw_config.idle_timeout_ms = config_.network.connection_timeout_ms;
160
161 gateway_ = std::make_unique<gateway::gateway_server>(gw_config);
162
163 // Set up connection callbacks for logging
164 gateway_->set_connection_callback(
165 [this](const gateway::client_session& session)
166 {
167 logger_->log(kcenon::common::interfaces::log_level::info,
168 "Client connected: " + session.session_id);
169 });
170
171 gateway_->set_disconnection_callback(
172 [this](const std::string& session_id)
173 {
174 logger_->log(kcenon::common::interfaces::log_level::info,
175 "Client disconnected: " + session_id);
176 });
177
178 // Set up request handler to route queries through query_router
179 gateway_->set_request_handler(
180 [this](const gateway::client_session& session, const gateway::query_request& request)
181 -> gateway::query_response
182 {
183 (void)session; // Session info available for future enhancements (logging, rate limiting)
184
185 // Execute query through query router
186 auto result = query_router_->execute(request);
187 if (result.is_ok())
188 {
189 return std::move(result.value());
190 }
191 else
192 {
193 return gateway::query_response(
194 request.header.message_id,
195 gateway::status_code::error,
196 result.error().message);
197 }
198 });
199
200 std::ostringstream init_msg;
201 init_msg << "Server '" << config_.name << "' initialized";
202 logger_->log(kcenon::common::interfaces::log_level::info, init_msg.str());
203
204 std::ostringstream addr_msg;
205 addr_msg << " Listen address: " << config_.network.host << ":" << config_.network.port;
206 logger_->log(kcenon::common::interfaces::log_level::info, addr_msg.str());
207
208 return true;
209}

211int server_app::run()
212{
213 if (state_ != server_state::initialized)
214 {
215 logger_->log(kcenon::common::interfaces::log_level::error,
216 std::string("Server not initialized"));
217 return 1;
218 }
219
220 state_ = server_state::starting;
221
222 // Start gateway server
223 if (gateway_)
224 {
225 auto result = gateway_->start();
226 if (result.is_err())
227 {
228 logger_->log(kcenon::common::interfaces::log_level::error,
229 "Failed to start gateway server: " + result.error().message);
230 state_ = server_state::stopped;
231 return 1;
232 }
233 }
234
235 // Start connection pool health monitoring if pool is configured
236 if (connection_pool_)
237 {
238 connection_pool_->schedule_health_check();
239 logger_->log(kcenon::common::interfaces::log_level::info,
240 std::string("Connection pool health monitoring started"));
241 }
242
243 state_ = server_state::running;
244 logger_->log(kcenon::common::interfaces::log_level::info,
245 std::string("Server is running. Press Ctrl+C to stop."));
246
247 // Main event loop - wait for shutdown signal
248 while (state_ == server_state::running)
249 {
250 std::this_thread::sleep_for(std::chrono::milliseconds(100));
251 }
252
253 state_ = server_state::stopped;
254 logger_->log(kcenon::common::interfaces::log_level::info,
255 std::string("Server stopped"));
256
257 return 0;
258}

260void server_app::stop()
261{
262 if (state_ != server_state::running)
263 {
264 return;
265 }
266
267 state_ = server_state::stopping;
268
269 // Stop gateway server
270 if (gateway_)
271 {
272 auto result = gateway_->stop();
273 if (result.is_err())
274 {
275 logger_->log(kcenon::common::interfaces::log_level::warning,
276 "Failed to stop gateway server: " + result.error().message);
277 }
278 }
279
280 // Shutdown connection pool
281 if (connection_pool_)
282 {
283 connection_pool_->request_shutdown();
284 connection_pool_->shutdown();
285 }
286
287 logger_->log(kcenon::common::interfaces::log_level::info,
288 std::string("Shutdown requested"));
289}

291void server_app::do_cleanup()
292{
293 // Cleanup query router
294 query_router_.reset();
295
296 // Cleanup connection pool
297 connection_pool_.reset();
298
299 // Cleanup gateway server
300 gateway_.reset();
301
302 // Shutdown executor if set
303 if (executor_)
304 {
305 executor_->shutdown(true);
306 executor_.reset();
307 }
308
309 if (instance_ == this)
310 {
311 instance_ = nullptr;
312 }

357void server_app::setup_signal_handlers()
358{
359 instance_ = this;
360
361#ifdef _WIN32
362 std::signal(SIGINT, signal_handler);
363 std::signal(SIGTERM, signal_handler);
364#else
365 struct sigaction sa;
366 sa.sa_handler = signal_handler;
367 sigemptyset(&sa.sa_mask);
368 sa.sa_flags = 0;
369
370 sigaction(SIGINT, &sa, nullptr);
371 sigaction(SIGTERM, &sa, nullptr);
372#endif
373}

375void server_app::signal_handler(int signal)
376{
377 if (instance_ != nullptr)
378 {
379 // Note: Signal handlers should be async-signal-safe.
380 // The logger_->log call in stop() is safe because stop() is called
381 // from the main thread context after the signal interrupts sleep.
382 // For truly async-safe logging, consider using write() directly.
383 (void)signal; // Signal number logged via stop() method
384 instance_->stop();
385 }
386}
