83struct pool_metrics
84{
85 // Connection statistics
86 std::atomic<uint64_t> total_acquisitions{ 0 };
87 std::atomic<uint64_t> successful_acquisitions{ 0 };
88 std::atomic<uint64_t> failed_acquisitions{ 0 };

90
91 // Timing statistics (microseconds)
92 std::atomic<uint64_t> total_wait_time_us{ 0 };
93 std::atomic<uint64_t> min_wait_time_us{ UINT64_MAX };
94 std::atomic<uint64_t> max_wait_time_us{ 0 };
95
96 // Current state
97 std::atomic<uint64_t> current_active{ 0 };
98 std::atomic<uint64_t> current_queued{ 0 };
99 std::atomic<uint64_t> peak_active{ 0 };
100 std::atomic<uint64_t> peak_queued{ 0 };
101
102 // Health check statistics
103 std::atomic<uint64_t> health_checks_performed{ 0 };
104 std::atomic<uint64_t> unhealthy_connections_removed{ 0 };

111 void record_acquisition(uint64_t wait_time_us, bool success)
112 {
113 total_acquisitions.fetch_add(1, std::memory_order_relaxed);
114
115 if (success)
116 {
117 successful_acquisitions.fetch_add(1, std::memory_order_relaxed);
118 total_wait_time_us.fetch_add(wait_time_us, std::memory_order_relaxed);
119 metrics_utils::update_min_max(min_wait_time_us, max_wait_time_us, wait_time_us);
120 }
121 else
122 {
123 failed_acquisitions.fetch_add(1, std::memory_order_relaxed);
124 }
125 }
126
130 void record_timeout()
131 {
132 timeouts.fetch_add(1, std::memory_order_relaxed);
133 }
134
139 void update_active(int delta)
140 {
141 uint64_t new_active
142 = current_active.fetch_add(delta, std::memory_order_relaxed) + delta;
143 metrics_utils::update_max(peak_active, new_active);
144 }
145
150 void update_queued(int delta)
151 {
152 uint64_t new_queued
153 = current_queued.fetch_add(delta, std::memory_order_relaxed) + delta;
154 metrics_utils::update_max(peak_queued, new_queued);
155 }
156
161 void record_health_check(uint64_t removed_connections = 0)
162 {
163 health_checks_performed.fetch_add(1, std::memory_order_relaxed);
164 if (removed_connections > 0)
165 {
166 unhealthy_connections_removed.fetch_add(
167 removed_connections, std::memory_order_relaxed);
168 }
169 }
170
175 [[nodiscard]] double average_wait_time_us() const
176 {
177 return metrics_utils::average_us(
178 total_wait_time_us.load(std::memory_order_relaxed),
179 total_acquisitions.load(std::memory_order_relaxed));
180 }
181
186 [[nodiscard]] double success_rate() const
187 {
188 return metrics_utils::calculate_rate(
189 successful_acquisitions.load(std::memory_order_relaxed),
190 total_acquisitions.load(std::memory_order_relaxed), 100.0);
191 }
192
196 void reset()
197 {
198 metrics_utils::reset_counter(total_acquisitions);
199 metrics_utils::reset_counter(successful_acquisitions);
200 metrics_utils::reset_counter(failed_acquisitions);
201 metrics_utils::reset_counter(timeouts);
202
203 metrics_utils::reset_counter(total_wait_time_us);
204 metrics_utils::reset_min(min_wait_time_us);
205 metrics_utils::reset_counter(max_wait_time_us);
206
207 // Don't reset current_active and current_queued (they reflect current state)
208 peak_active.store(
209 current_active.load(std::memory_order_relaxed), std::memory_order_relaxed);
210 peak_queued.store(
211 current_queued.load(std::memory_order_relaxed), std::memory_order_relaxed);
212
213 metrics_utils::reset_counter(health_checks_performed);
214 metrics_utils::reset_counter(unhealthy_connections_removed);
215 }
216};

225template <typename PriorityType>
226struct priority_metrics : public pool_metrics
227{
228 using priority_type = PriorityType;
229
230 // Per-priority statistics
231 std::map<PriorityType, std::atomic<uint64_t>> acquisitions_by_priority;
232 std::map<PriorityType, std::atomic<uint64_t>> wait_time_by_priority;
233
234 mutable std::mutex map_mutex; // Protects map modifications
235
242 void record_acquisition_with_priority(
243 PriorityType priority, uint64_t wait_time_us, bool success)
244 {
245 // Record base metrics
246 record_acquisition(wait_time_us, success);
247
248 if (success)
249 {
250 // Update priority-specific metrics
251 std::lock_guard<std::mutex> lock(map_mutex);
252
253 // Initialize if first time seeing this priority
254 if (acquisitions_by_priority.find(priority)
255 == acquisitions_by_priority.end())
256 {
257 acquisitions_by_priority[priority].store(0, std::memory_order_relaxed);
258 wait_time_by_priority[priority].store(0, std::memory_order_relaxed);
259 }
260
261 acquisitions_by_priority[priority].fetch_add(1, std::memory_order_relaxed);
262 wait_time_by_priority[priority].fetch_add(
263 wait_time_us, std::memory_order_relaxed);
264 }
265 }
266
272 [[nodiscard]] double average_wait_time_for_priority(PriorityType priority) const
273 {
274 std::lock_guard<std::mutex> lock(map_mutex);
275
276 auto acq_it = acquisitions_by_priority.find(priority);
277 auto wait_it = wait_time_by_priority.find(priority);
278
279 if (acq_it == acquisitions_by_priority.end()
280 || wait_it == wait_time_by_priority.end())
281 {
282 return 0.0;
283 }
284
285 return metrics_utils::average_us(
286 wait_it->second.load(std::memory_order_relaxed),
287 acq_it->second.load(std::memory_order_relaxed));
288 }
289
293 void reset_all()
294 {
295 pool_metrics::reset();
296
297 std::lock_guard<std::mutex> lock(map_mutex);
298 for (auto& [priority, counter] : acquisitions_by_priority)
299 {
300 metrics_utils::reset_counter(counter);
301 }
302 for (auto& [priority, wait_time] : wait_time_by_priority)
303 {
304 metrics_utils::reset_counter(wait_time);
305 }
306 }
307};
