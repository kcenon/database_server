66struct pool_metrics
67{
68 // Connection statistics
69 std::atomic<uint64_t> total_acquisitions{ 0 };
70 std::atomic<uint64_t> successful_acquisitions{ 0 };
71 std::atomic<uint64_t> failed_acquisitions{ 0 };
72 std::atomic<uint64_t> timeouts{ 0 };
73
74 // Timing statistics (microseconds)
75 std::atomic<uint64_t> total_wait_time_us{ 0 };
76 std::atomic<uint64_t> min_wait_time_us{ UINT64_MAX };
77 std::atomic<uint64_t> max_wait_time_us{ 0 };
78
79 // Current state
80 std::atomic<uint64_t> current_active{ 0 };
81 std::atomic<uint64_t> current_queued{ 0 };
82 std::atomic<uint64_t> peak_active{ 0 };
83 std::atomic<uint64_t> peak_queued{ 0 };
84
85 // Health check statistics
86 std::atomic<uint64_t> health_checks_performed{ 0 };
87 std::atomic<uint64_t> unhealthy_connections_removed{ 0 };
88
94 void record_acquisition(uint64_t wait_time_us, bool success)
95 {
96 total_acquisitions.fetch_add(1, std::memory_order_relaxed);
97
98 if (success)
99 {
100 successful_acquisitions.fetch_add(1, std::memory_order_relaxed);
101 total_wait_time_us.fetch_add(wait_time_us, std::memory_order_relaxed);
102 metrics_utils::update_min_max(min_wait_time_us, max_wait_time_us, wait_time_us);
103 }
104 else
105 {
106 failed_acquisitions.fetch_add(1, std::memory_order_relaxed);
107 }
108 }
109
113 void record_timeout()
114 {
115 timeouts.fetch_add(1, std::memory_order_relaxed);
116 }
117
122 void update_active(int delta)
123 {
124 uint64_t new_active
125 = current_active.fetch_add(delta, std::memory_order_relaxed) + delta;
126 metrics_utils::update_max(peak_active, new_active);
127 }
128
133 void update_queued(int delta)
134 {
135 uint64_t new_queued
136 = current_queued.fetch_add(delta, std::memory_order_relaxed) + delta;
137 metrics_utils::update_max(peak_queued, new_queued);
138 }
139
144 void record_health_check(uint64_t removed_connections = 0)
145 {
146 health_checks_performed.fetch_add(1, std::memory_order_relaxed);
147 if (removed_connections > 0)
148 {
149 unhealthy_connections_removed.fetch_add(
150 removed_connections, std::memory_order_relaxed);
151 }
152 }
153
158 [[nodiscard]] double average_wait_time_us() const
159 {
160 return metrics_utils::average_us(
161 total_wait_time_us.load(std::memory_order_relaxed),
162 total_acquisitions.load(std::memory_order_relaxed));
163 }
164
169 [[nodiscard]] double success_rate() const
170 {
171 return metrics_utils::calculate_rate(
172 successful_acquisitions.load(std::memory_order_relaxed),
173 total_acquisitions.load(std::memory_order_relaxed), 100.0);
174 }
175
179 void reset()
180 {
181 metrics_utils::reset_counter(total_acquisitions);
182 metrics_utils::reset_counter(successful_acquisitions);
183 metrics_utils::reset_counter(failed_acquisitions);
184 metrics_utils::reset_counter(timeouts);
185
186 metrics_utils::reset_counter(total_wait_time_us);
187 metrics_utils::reset_min(min_wait_time_us);
188 metrics_utils::reset_counter(max_wait_time_us);
189
190 // Don't reset current_active and current_queued (they reflect current state)
191 peak_active.store(
192 current_active.load(std::memory_order_relaxed), std::memory_order_relaxed);
193 peak_queued.store(
194 current_queued.load(std::memory_order_relaxed), std::memory_order_relaxed);
195
196 metrics_utils::reset_counter(health_checks_performed);
197 metrics_utils::reset_counter(unhealthy_connections_removed);
198 }
199};

208template <typename PriorityType>
209struct priority_metrics : public pool_metrics
210{
211 using priority_type = PriorityType;
212
213 // Per-priority statistics
214 std::map<PriorityType, std::atomic<uint64_t>> acquisitions_by_priority;
215 std::map<PriorityType, std::atomic<uint64_t>> wait_time_by_priority;
216
217 mutable std::mutex map_mutex; // Protects map modifications
218
225 void record_acquisition_with_priority(
226 PriorityType priority, uint64_t wait_time_us, bool success)
227 {
228 // Record base metrics
229 record_acquisition(wait_time_us, success);
230
231 if (success)
232 {
233 // Update priority-specific metrics
234 std::lock_guard<std::mutex> lock(map_mutex);
235
236 // Initialize if first time seeing this priority
237 if (acquisitions_by_priority.find(priority)
238 == acquisitions_by_priority.end())
239 {
240 acquisitions_by_priority[priority].store(0, std::memory_order_relaxed);
241 wait_time_by_priority[priority].store(0, std::memory_order_relaxed);
242 }
243
244 acquisitions_by_priority[priority].fetch_add(1, std::memory_order_relaxed);
245 wait_time_by_priority[priority].fetch_add(
246 wait_time_us, std::memory_order_relaxed);
247 }
248 }
249
255 [[nodiscard]] double average_wait_time_for_priority(PriorityType priority) const
256 {
257 std::lock_guard<std::mutex> lock(map_mutex);
258
259 auto acq_it = acquisitions_by_priority.find(priority);
260 auto wait_it = wait_time_by_priority.find(priority);
261
262 if (acq_it == acquisitions_by_priority.end()
263 || wait_it == wait_time_by_priority.end())
264 {
265 return 0.0;
266 }
267
268 return metrics_utils::average_us(
269 wait_it->second.load(std::memory_order_relaxed),
270 acq_it->second.load(std::memory_order_relaxed));
271 }
272
276 void reset_all()
277 {
278 pool_metrics::reset();
279
280 std::lock_guard<std::mutex> lock(map_mutex);
281 for (auto& [priority, counter] : acquisitions_by_priority)
282 {
283 metrics_utils::reset_counter(counter);
284 }
285 for (auto& [priority, wait_time] : wait_time_by_priority)
286 {
287 metrics_utils::reset_counter(wait_time);
288 }
289 }
290};
