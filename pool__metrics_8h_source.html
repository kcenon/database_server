62struct pool_metrics
63{
64 // Connection statistics
65 std::atomic<uint64_t> total_acquisitions{ 0 };
66 std::atomic<uint64_t> successful_acquisitions{ 0 };
67 std::atomic<uint64_t> failed_acquisitions{ 0 };
68 std::atomic<uint64_t> timeouts{ 0 };
69
70 // Timing statistics (microseconds)
71 std::atomic<uint64_t> total_wait_time_us{ 0 };
72 std::atomic<uint64_t> min_wait_time_us{ UINT64_MAX };
73 std::atomic<uint64_t> max_wait_time_us{ 0 };
74
75 // Current state
76 std::atomic<uint64_t> current_active{ 0 };
77 std::atomic<uint64_t> current_queued{ 0 };
78 std::atomic<uint64_t> peak_active{ 0 };
79 std::atomic<uint64_t> peak_queued{ 0 };
80
81 // Health check statistics
82 std::atomic<uint64_t> health_checks_performed{ 0 };
83 std::atomic<uint64_t> unhealthy_connections_removed{ 0 };
84
90 void record_acquisition(uint64_t wait_time_us, bool success)
91 {
92 total_acquisitions.fetch_add(1, std::memory_order_relaxed);
93
94 if (success)
95 {
96 successful_acquisitions.fetch_add(1, std::memory_order_relaxed);
97
98 // Update timing statistics
99 total_wait_time_us.fetch_add(wait_time_us, std::memory_order_relaxed);
100
101 // Update min/max (requires atomic compare-exchange)
102 uint64_t current_min = min_wait_time_us.load(std::memory_order_relaxed);
103 while (wait_time_us < current_min
104 && !min_wait_time_us.compare_exchange_weak(
105 current_min, wait_time_us, std::memory_order_relaxed))
106 {
107 }
108
109 uint64_t current_max = max_wait_time_us.load(std::memory_order_relaxed);
110 while (wait_time_us > current_max
111 && !max_wait_time_us.compare_exchange_weak(
112 current_max, wait_time_us, std::memory_order_relaxed))
113 {
114 }
115 }
116 else
117 {
118 failed_acquisitions.fetch_add(1, std::memory_order_relaxed);
119 }
120 }
121
125 void record_timeout()
126 {
127 timeouts.fetch_add(1, std::memory_order_relaxed);
128 }
129
134 void update_active(int delta)
135 {
136 uint64_t new_active
137 = current_active.fetch_add(delta, std::memory_order_relaxed) + delta;
138
139 // Update peak if necessary
140 uint64_t current_peak = peak_active.load(std::memory_order_relaxed);
141 while (new_active > current_peak
142 && !peak_active.compare_exchange_weak(
143 current_peak, new_active, std::memory_order_relaxed))
144 {
145 }
146 }
147
152 void update_queued(int delta)
153 {
154 uint64_t new_queued
155 = current_queued.fetch_add(delta, std::memory_order_relaxed) + delta;
156
157 // Update peak if necessary
158 uint64_t current_peak = peak_queued.load(std::memory_order_relaxed);
159 while (new_queued > current_peak
160 && !peak_queued.compare_exchange_weak(
161 current_peak, new_queued, std::memory_order_relaxed))
162 {
163 }
164 }
165
170 void record_health_check(uint64_t removed_connections = 0)
171 {
172 health_checks_performed.fetch_add(1, std::memory_order_relaxed);
173 if (removed_connections > 0)
174 {
175 unhealthy_connections_removed.fetch_add(
176 removed_connections, std::memory_order_relaxed);
177 }
178 }
179
184 [[nodiscard]] double average_wait_time_us() const
185 {
186 uint64_t total = total_acquisitions.load(std::memory_order_relaxed);
187 if (total == 0)
188 return 0.0;
189
190 uint64_t total_wait = total_wait_time_us.load(std::memory_order_relaxed);
191 return static_cast<double>(total_wait) / static_cast<double>(total);
192 }
193
198 [[nodiscard]] double success_rate() const
199 {
200 uint64_t total = total_acquisitions.load(std::memory_order_relaxed);
201 if (total == 0)
202 return 100.0;
203
204 uint64_t successful = successful_acquisitions.load(std::memory_order_relaxed);
205 return (static_cast<double>(successful) / static_cast<double>(total)) * 100.0;
206 }
207
211 void reset()
212 {
213 total_acquisitions.store(0, std::memory_order_relaxed);
214 successful_acquisitions.store(0, std::memory_order_relaxed);
215 failed_acquisitions.store(0, std::memory_order_relaxed);
216 timeouts.store(0, std::memory_order_relaxed);
217
218 total_wait_time_us.store(0, std::memory_order_relaxed);
219 min_wait_time_us.store(UINT64_MAX, std::memory_order_relaxed);
220 max_wait_time_us.store(0, std::memory_order_relaxed);
221
222 // Don't reset current_active and current_queued (they reflect current state)
223 peak_active.store(
224 current_active.load(std::memory_order_relaxed), std::memory_order_relaxed);
225 peak_queued.store(
226 current_queued.load(std::memory_order_relaxed), std::memory_order_relaxed);
227
228 health_checks_performed.store(0, std::memory_order_relaxed);
229 unhealthy_connections_removed.store(0, std::memory_order_relaxed);
230 }
231};

240template <typename PriorityType>
241struct priority_metrics : public pool_metrics
242{
243 using priority_type = PriorityType;
244
245 // Per-priority statistics
246 std::map<PriorityType, std::atomic<uint64_t>> acquisitions_by_priority;
247 std::map<PriorityType, std::atomic<uint64_t>> wait_time_by_priority;
248
249 mutable std::mutex map_mutex; // Protects map modifications
250
257 void record_acquisition_with_priority(
258 PriorityType priority, uint64_t wait_time_us, bool success)
259 {
260 // Record base metrics
261 record_acquisition(wait_time_us, success);
262
263 if (success)
264 {
265 // Update priority-specific metrics
266 std::lock_guard<std::mutex> lock(map_mutex);
267
268 // Initialize if first time seeing this priority
269 if (acquisitions_by_priority.find(priority)
270 == acquisitions_by_priority.end())
271 {
272 acquisitions_by_priority[priority].store(0, std::memory_order_relaxed);
273 wait_time_by_priority[priority].store(0, std::memory_order_relaxed);
274 }
275
276 acquisitions_by_priority[priority].fetch_add(1, std::memory_order_relaxed);
277 wait_time_by_priority[priority].fetch_add(
278 wait_time_us, std::memory_order_relaxed);
279 }
280 }
281
287 [[nodiscard]] double average_wait_time_for_priority(PriorityType priority) const
288 {
289 std::lock_guard<std::mutex> lock(map_mutex);
290
291 auto acq_it = acquisitions_by_priority.find(priority);
292 auto wait_it = wait_time_by_priority.find(priority);
293
294 if (acq_it == acquisitions_by_priority.end()
295 || wait_it == wait_time_by_priority.end())
296 {
297 return 0.0;
298 }
299
300 uint64_t count = acq_it->second.load(std::memory_order_relaxed);
301 if (count == 0)
302 return 0.0;
303
304 uint64_t total_wait = wait_it->second.load(std::memory_order_relaxed);
305 return static_cast<double>(total_wait) / static_cast<double>(count);
306 }
307
311 void reset_all()
312 {
313 pool_metrics::reset();
314
315 std::lock_guard<std::mutex> lock(map_mutex);
316 for (auto& [priority, counter] : acquisitions_by_priority)
317 {
318 counter.store(0, std::memory_order_relaxed);
319 }
320 for (auto& [priority, wait_time] : wait_time_by_priority)
321 {
322 wait_time.store(0, std::memory_order_relaxed);
323 }
324 }
325};
