41std::optional<server_config> server_config::load_from_file(const std::string& path)
42{
43 if (!std::filesystem::exists(path))
44 {
45 // Error will be logged by caller with appropriate context
46 return std::nullopt;
47 }
48
49 std::ifstream file(path);
50 if (!file.is_open())
51 {
52 // Error will be logged by caller with appropriate context
53 return std::nullopt;
54 }
55
56 // For Phase 1, we use a simple key=value format
57 // TODO: Add YAML parsing in future phases (using yaml-cpp or similar)
58 server_config config = default_config();
59
60 std::string line;
61 while (std::getline(file, line))
62 {
63 // Skip comments and empty lines
64 if (line.empty() || line[0] == '#')
65 {
66 continue;
67 }
68
69 auto delimiter_pos = line.find('=');
70 if (delimiter_pos == std::string::npos)
71 {
72 continue;
73 }
74
75 std::string key = line.substr(0, delimiter_pos);
76 std::string value = line.substr(delimiter_pos + 1);
77
78 // Trim whitespace
79 auto trim = [](std::string& s)
80 {
81 s.erase(0, s.find_first_not_of(" \t\r\n"));
82 s.erase(s.find_last_not_of(" \t\r\n") + 1);
83 };
84 trim(key);
85 trim(value);
86
87 // Parse configuration values
88 if (key == "name")
89 {
90 config.name = value;
91 }
92 else if (key == "network.host")
93 {
94 config.network.host = value;
95 }
96 else if (key == "network.port")
97 {
98 config.network.port = static_cast<uint16_t>(std::stoi(value));
99 }
100 else if (key == "network.enable_tls")
101 {
102 config.network.enable_tls = (value == "true" || value == "1");
103 }
104 else if (key == "network.cert_file")
105 {
106 config.network.cert_file = value;
107 }
108 else if (key == "network.key_file")
109 {
110 config.network.key_file = value;
111 }
112 else if (key == "network.max_connections")
113 {
114 config.network.max_connections = static_cast<uint32_t>(std::stoul(value));
115 }
116 else if (key == "network.connection_timeout_ms")
117 {
118 config.network.connection_timeout_ms = static_cast<uint32_t>(std::stoul(value));
119 }
120 else if (key == "logging.level")
121 {
122 config.logging.level = value;
123 }
124 else if (key == "logging.log_file")
125 {
126 config.logging.log_file = value;
127 }
128 else if (key == "logging.enable_console")
129 {
130 config.logging.enable_console = (value == "true" || value == "1");
131 }
132 else if (key == "pool.min_connections")
133 {
134 config.pool.min_connections = static_cast<uint32_t>(std::stoul(value));
135 }
136 else if (key == "pool.max_connections")
137 {
138 config.pool.max_connections = static_cast<uint32_t>(std::stoul(value));
139 }
140 else if (key == "pool.idle_timeout_ms")
141 {
142 config.pool.idle_timeout_ms = static_cast<uint32_t>(std::stoul(value));
143 }
144 else if (key == "pool.health_check_interval_ms")
145 {
146 config.pool.health_check_interval_ms = static_cast<uint32_t>(std::stoul(value));
147 }
148 }
149
150 return config;
151}

165std::vector<std::string> server_config::validation_errors() const
166{
167 std::vector<std::string> errors;
168
169 // Validate name
170 if (name.empty())
171 {
172 errors.push_back("Server name cannot be empty");
173 }
174
175 // Validate network configuration
176 if (network.host.empty())
177 {
178 errors.push_back("Network host cannot be empty");
179 }
180
181 if (network.port == 0)
182 {
183 errors.push_back("Network port must be greater than 0");
184 }
185
186 if (network.enable_tls)
187 {
188 if (network.cert_file.empty())
189 {
190 errors.push_back("TLS is enabled but no certificate file specified");
191 }
192 else if (!std::filesystem::exists(network.cert_file))
193 {
194 errors.push_back("TLS certificate file not found: " + network.cert_file);
195 }
196
197 if (network.key_file.empty())
198 {
199 errors.push_back("TLS is enabled but no key file specified");
200 }
201 else if (!std::filesystem::exists(network.key_file))
202 {
203 errors.push_back("TLS key file not found: " + network.key_file);
204 }
205 }
206
207 if (network.max_connections == 0)
208 {
209 errors.push_back("Maximum connections must be greater than 0");
210 }
211
212 // Validate pool configuration
213 if (pool.min_connections > pool.max_connections)
214 {
215 errors.push_back("Pool minimum connections cannot exceed maximum connections");
216 }
217
218 // Validate logging configuration
219 if (logging.level != "debug" && logging.level != "info" && logging.level != "warn"
220 && logging.level != "error")
221 {
222 errors.push_back("Invalid log level: " + logging.level
223 + " (valid: debug, info, warn, error)");
224 }
225
226 return errors;
227}
