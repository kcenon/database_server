86struct cache_metrics
87{
88 std::atomic<uint64_t> hits{0};
89 std::atomic<uint64_t> misses{0};
90 std::atomic<uint64_t> evictions{0};
91 std::atomic<uint64_t> expirations{0};
92 std::atomic<uint64_t> invalidations{0};
93 std::atomic<uint64_t> puts{0};
94 std::atomic<uint64_t> skipped_too_large{0};
95
100 [[nodiscard]] double hit_rate() const noexcept
101 {
102 auto total = hits.load() + misses.load();
103 if (total == 0) return 0.0;
104 return static_cast<double>(hits.load()) / static_cast<double>(total) * 100.0;
105 }
106
110 void reset() noexcept
111 {
112 hits.store(0);
113 misses.store(0);
114 evictions.store(0);
115 expirations.store(0);
116 invalidations.store(0);
117 puts.store(0);
118 skipped_too_large.store(0);
119 }
120};
