75class connection_acquisition_job
76 : public kcenon::thread::typed_job_t<connection_priority>
77{
78public:
79 using completion_callback = std::function<void(
80 kcenon::common::Result<std::shared_ptr<database::connection_wrapper>>)>;
81
88 explicit connection_acquisition_job(
89 connection_priority priority,
90 std::shared_ptr<database::connection_pool_base> pool_ref,
91 completion_callback callback)
92 : kcenon::thread::typed_job_t<connection_priority>(
93 priority, "connection_acquisition")
94 , pool_ref_(std::move(pool_ref))
95 , callback_(std::move(callback))
96 {
97 }
98
99 kcenon::common::VoidResult do_work() override
100 {
101 try
102 {
103 // Acquire connection from the underlying pool
104 auto result = pool_ref_->acquire_connection();
105
106 // Invoke callback with result
107 if (callback_)
108 {
109 callback_(std::move(result));
110 }
111
112 return kcenon::common::ok();
113 }
114 catch (const std::exception& e)
115 {
116 // On exception, invoke callback with error
117 if (callback_)
118 {
119 callback_(kcenon::common::error_info{
120 -599,
121 std::string("Exception in connection acquisition: ") + e.what(),
122 "connection_pool" });
123 }
124
125 return kcenon::common::error_info{
126 static_cast<int>(kcenon::thread::error_code::job_execution_failed),
127 std::string("Exception in connection_acquisition_job: ") + e.what(),
128 "connection_pool"
129 };
130 }
131 }
132
133private:
134 std::shared_ptr<database::connection_pool_base> pool_ref_;
135 completion_callback callback_;
136};
