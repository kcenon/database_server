78class connection_acquisition_job
79 : public kcenon::thread::typed_job_t<connection_priority>
80{
81public:
82 using completion_callback = std::function<void(
83 kcenon::common::Result<std::shared_ptr<database::connection_wrapper>>)>;
84
91 explicit connection_acquisition_job(
92 connection_priority priority,
93 std::shared_ptr<database::connection_pool_base> pool_ref,
94 completion_callback callback)
95 : kcenon::thread::typed_job_t<connection_priority>(
96 priority, "connection_acquisition")
97 , pool_ref_(std::move(pool_ref))
98 , callback_(std::move(callback))
99 {
100 }
101
102 kcenon::common::VoidResult do_work() override
103 {
104 try
105 {
106 // Acquire connection from the underlying pool
107 auto result = pool_ref_->acquire_connection();
108
109 // Invoke callback with result
110 if (callback_)
111 {
112 callback_(std::move(result));
113 }
114
115 return kcenon::common::ok();
116 }
117 catch (const std::exception& e)
118 {
119 // On exception, invoke callback with error
120 if (callback_)
121 {
122 callback_(kcenon::common::error_info{
123 -599,
124 std::string("Exception in connection acquisition: ") + e.what(),
125 "connection_pool" });
126 }
127
128 return kcenon::common::error_info{
129 static_cast<int>(kcenon::thread::error_code::job_execution_failed),
130 std::string("Exception in connection_acquisition_job: ") + e.what(),
131 "connection_pool"
132 };
133 }
134 }
135
136private:
137 std::shared_ptr<database::connection_pool_base> pool_ref_;
138 completion_callback callback_;
139};
