233class connection_pool : public connection_pool_base
234{
235public:
236 using connection_factory = std::function<std::unique_ptr<core::database_backend>()>;
237
238 connection_pool(
239 database_types db_type,
240 const connection_pool_config& config,
241 connection_factory factory)
242 : db_type_(db_type)
243 , config_(config)
244 , factory_(std::move(factory))
245 , shutting_down_(false)
246 {
247 }
248
249 ~connection_pool() override
250 {
251 shutdown();
252 }
253
254 bool initialize()
255 {
256 std::lock_guard<std::mutex> lock(pool_mutex_);
257
258 for (size_t i = 0; i < config_.min_connections; ++i)
259 {
260 auto conn = create_connection();
261 if (conn)
262 {
263 available_connections_.push(std::move(conn));
264 stats_.total_connections++;
265 }
266 }
267
268 stats_.available_connections = available_connections_.size();
269 return stats_.available_connections > 0;
270 }
271
272 kcenon::common::Result<std::shared_ptr<connection_wrapper>> acquire_connection() override
273 {
274 if (shutting_down_.load())
275 {
276 return kcenon::common::error_info{-500, "Pool is shutting down", "connection_pool"};
277 }
278
279 std::unique_lock<std::mutex> lock(pool_mutex_);
280
281 // Wait for available connection or timeout
282 auto deadline = std::chrono::steady_clock::now() + config_.acquire_timeout;
283
284 while (available_connections_.empty())
285 {
286 // Try to create new connection if under max
287 if (stats_.total_connections < config_.max_connections)
288 {
289 lock.unlock();
290 auto conn = create_connection();
291 lock.lock();
292
293 if (conn)
294 {
295 stats_.total_connections++;
296 stats_.active_connections++;
297 stats_.successful_acquisitions++;
298 return conn;
299 }
300 }
301
302 // Wait for connection to be released
303 if (pool_condition_.wait_until(lock, deadline) == std::cv_status::timeout)
304 {
305 stats_.failed_acquisitions++;
306 return kcenon::common::error_info{-501, "Connection acquisition timeout", "connection_pool"};
307 }
308
309 if (shutting_down_.load())
310 {
311 return kcenon::common::error_info{-500, "Pool is shutting down", "connection_pool"};
312 }
313 }
314
315 auto conn = std::move(available_connections_.front());
316 available_connections_.pop();
317 stats_.available_connections = available_connections_.size();
318 stats_.active_connections++;
319 stats_.successful_acquisitions++;
320
321 conn->update_last_used();
322 return conn;
323 }
324
325 void release_connection(std::shared_ptr<connection_wrapper> connection) override
326 {
327 if (!connection)
328 {
329 return;
330 }
331
332 std::lock_guard<std::mutex> lock(pool_mutex_);
333
334 if (shutting_down_.load())
335 {
336 stats_.active_connections--;
337 return;
338 }
339
340 if (connection->is_healthy())
341 {
342 available_connections_.push(std::move(connection));
343 stats_.available_connections = available_connections_.size();
344 }
345
346 stats_.active_connections--;
347 pool_condition_.notify_one();
348 }
349
350 connection_stats get_stats() const override
351 {
352 std::lock_guard<std::mutex> lock(pool_mutex_);
353 return stats_;
354 }
355
356 size_t active_connections() const override
357 {
358 std::lock_guard<std::mutex> lock(pool_mutex_);
359 return stats_.active_connections;
360 }
361
362 size_t available_connections() const override
363 {
364 std::lock_guard<std::mutex> lock(pool_mutex_);
365 return stats_.available_connections;
366 }
367
368 void health_check() override
369 {
370 std::lock_guard<std::mutex> lock(pool_mutex_);
371 stats_.last_health_check = std::chrono::steady_clock::now();
372
373 // Check and remove unhealthy connections
374 std::queue<std::shared_ptr<connection_wrapper>> healthy_connections;
375 while (!available_connections_.empty())
376 {
377 auto conn = std::move(available_connections_.front());
378 available_connections_.pop();
379
380 if (conn->is_healthy() && !conn->is_idle_timeout_exceeded(config_.idle_timeout))
381 {
382 healthy_connections.push(std::move(conn));
383 }
384 else
385 {
386 stats_.total_connections--;
387 }
388 }
389 available_connections_ = std::move(healthy_connections);
390 stats_.available_connections = available_connections_.size();
391 }
392
393 void shutdown() override
394 {
395 shutting_down_.store(true);
396 pool_condition_.notify_all();
397
398 std::lock_guard<std::mutex> lock(pool_mutex_);
399 while (!available_connections_.empty())
400 {
401 available_connections_.pop();
402 }
403 stats_.available_connections = 0;
404 }
405
406 bool is_shutting_down() const override
407 {
408 return shutting_down_.load();
409 }
410
411private:
412 std::shared_ptr<connection_wrapper> create_connection()
413 {
414 try
415 {
416 auto db = factory_();
417 if (db)
418 {
419 return std::make_shared<connection_wrapper>(std::move(db));
420 }
421 }
422 catch (...)
423 {
424 // Failed to create connection
425 }
426 return nullptr;
427 }

429 database_types db_type_;
430 connection_pool_config config_;
431 connection_factory factory_;
432
433 mutable std::mutex pool_mutex_;
434 std::condition_variable pool_condition_;
435 std::queue<std::shared_ptr<connection_wrapper>> available_connections_;
436 connection_stats stats_;
437 std::atomic<bool> shutting_down_;
438};
