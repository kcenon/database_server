61struct health_status
62{
63 bool is_healthy{ false };
64 uint32_t health_score{ 0 }; // 0-100 scale
65 std::chrono::milliseconds latency{ 0 };
66 uint64_t successful_queries{ 0 };
67 uint64_t failed_queries{ 0 };
68 std::chrono::system_clock::time_point last_check_time;
69 std::string status_message;
70
71 [[nodiscard]] double success_rate() const
72 {
73 uint64_t total = successful_queries + failed_queries;
74 return total > 0
75 ? (static_cast<double>(successful_queries) / total) * 100.0
76 : 100.0;
77 }
78};
