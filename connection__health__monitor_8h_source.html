64struct health_status
65{
66 bool is_healthy{ false };
67 uint32_t health_score{ 0 }; // 0-100 scale
68 std::chrono::milliseconds latency{ 0 };
69 uint64_t successful_queries{ 0 };
70 uint64_t failed_queries{ 0 };
71 std::chrono::system_clock::time_point last_check_time;
72 std::string status_message;
73
74 [[nodiscard]] double success_rate() const
75 {
76 uint64_t total = successful_queries + failed_queries;
77 return total > 0
78 ? (static_cast<double>(successful_queries) / total) * 100.0
79 : 100.0;
80 }
81};
