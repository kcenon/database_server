173inline query_type parse_query_type(std::string_view str) noexcept
174{
175 if (str == "SELECT" || str == "select")
176 return query_type::select;
177 if (str == "INSERT" || str == "insert")
178 return query_type::insert;
179 if (str == "UPDATE" || str == "update")
180 return query_type::update;
181 if (str == "DELETE" || str == "delete")
182 return query_type::del;
183 if (str == "EXECUTE" || str == "execute")
184 return query_type::execute;
185 if (str == "BATCH" || str == "batch")
186 return query_type::batch;
187 if (str == "PING" || str == "ping")
188 return query_type::ping;
189 return query_type::unknown;
190}
