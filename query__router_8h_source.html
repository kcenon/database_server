93struct router_metrics
94{
95 std::atomic<uint64_t> total_queries{0};
96 std::atomic<uint64_t> successful_queries{0};
97 std::atomic<uint64_t> failed_queries{0};
98 std::atomic<uint64_t> timeout_queries{0};
99 std::atomic<uint64_t> total_execution_time_us{0};
100
101 double average_execution_time_us() const noexcept
102 {
103 auto total = total_queries.load();
104 if (total == 0) return 0.0;
105 return static_cast<double>(total_execution_time_us.load()) / total;
106 }
107
108 double success_rate() const noexcept
109 {
110 auto total = total_queries.load();
111 if (total == 0) return 0.0;
112 return static_cast<double>(successful_queries.load()) / total * 100.0;
113 }
114};
