87struct router_metrics
88{
89 std::atomic<uint64_t> total_queries{0};
90 std::atomic<uint64_t> successful_queries{0};
91 std::atomic<uint64_t> failed_queries{0};
92 std::atomic<uint64_t> timeout_queries{0};
93 std::atomic<uint64_t> total_execution_time_us{0};
94
95 double average_execution_time_us() const noexcept
96 {
97 auto total = total_queries.load();
98 if (total == 0) return 0.0;
99 return static_cast<double>(total_execution_time_us.load()) / total;
100 }
101
102 double success_rate() const noexcept
103 {
104 auto total = total_queries.load();
105 if (total == 0) return 0.0;
106 return static_cast<double>(successful_queries.load()) / total * 100.0;
107 }
108};
